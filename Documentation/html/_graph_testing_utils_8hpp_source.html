179  template<maths::graph_flavour F, class C, class E=typename C::edge_init_type>
180  void check_graph_edges(const C& connectivity, const std::size_t size, std::initializer_list<std::initializer_list<E>> edges, std::string_view failureMessage="")
181  {
182  using Edge = typename C::edge_type;
183  using RefEdge = E;
184  constexpr bool nullEdgeWeight{std::is_empty_v<typename C::edge_weight_type>};
185  double nEdges{};
186  for(auto edgesIter{edges.begin()}; edgesIter != edges.end(); ++edgesIter)
187  {
188  const auto i{std::distance(edges.begin(), edgesIter)};
189  const std::string partStr{std::to_string(i)};
190 
191  auto beginEdges{connectivity.cbegin_edges(i)};
192  auto endEdges{connectivity.cend_edges(i)};
193  auto rbeginEdges{connectivity.crbegin_edges(i)};
194  auto rendEdges{connectivity.crend_edges(i)};
195 
196  const auto nNodeEdges{static_cast<std::size_t>(distance(beginEdges, endEdges))};
197  const auto nrNodeEdges{static_cast<std::size_t>(distance(rbeginEdges, rendEdges))};
198 
199  const auto nodeEdges{*edgesIter};
200  if(check_equality(nodeEdges.size(), nNodeEdges, "Number of edges for partition " + partStr + " wrong")
201  && check_equality(nodeEdges.size(), nrNodeEdges, "Number of edges for reversed partition " + partStr + " wrong")
202  )
203  {
204  auto ansIter{nodeEdges.begin()};
205  auto redge{rendEdges};
206  for(auto edge{beginEdges}; edge != endEdges; ++edge, ++ansIter)
207  {
208  --redge;
209  check(*edge == *redge, "Disagreement between forward and reverse itereators");
210  auto dist{distance(beginEdges, edge)};
211  const std::string message{" wrong for partition " + partStr + ", edge " + to_string(dist)};
212 
213  const auto target{[edge=*ansIter](const std::size_t node){
214  if constexpr ( (F != maths::graph_flavour::directed_embedded)
215  && (E::flavour != maths::edge_flavour::partial)
216  && (E::flavour != maths::edge_flavour::partial_embedded))
217  {
218  return (edge.host_node() == node) ? edge.target_node() : edge.host_node();
219  }
220  else
221  {
222  return edge.target_node();
223  }
224  }(i)};
225 
226  check_equality(target, edge->target_node(), "target_node" + message);
227 
228  if constexpr ( (Edge::flavour == maths::edge_flavour::partial_embedded)
229  && ( (RefEdge::flavour == maths::edge_flavour::full_embedded)
230  || (RefEdge::flavour == maths::edge_flavour::partial_embedded)))
231  {
232  check_equality(ansIter->complementary_index(), edge->complementary_index(), "complementary_index" + message);
233  }
234  else if constexpr ((Edge::flavour == maths::edge_flavour::full_embedded) && (RefEdge::flavour == maths::edge_flavour::full_embedded))
235  {
236  check_equality(ansIter->host_node(), edge->host_node(), "host_node " + message);
237  check_equality(ansIter->complementary_index(), edge->complementary_index(), "complementary_index" + message);
238  check_equality(ansIter->inverted(), edge->inverted(), "inverted" + message);
239  }
240  else if constexpr ((Edge::flavour == maths::edge_flavour::full) && (RefEdge::flavour == maths::edge_flavour::full))
241  {
242  check_equality(ansIter->host_node(), edge->host_node(), "host_node" + message);
243  check_equality(ansIter->inverted(), edge->inverted(), "inverted" + message);
244  }
245 
246  if constexpr(!nullEdgeWeight) check_equality(ansIter->weight(), edge->weight(), "Weight" + message);
247 
248  (F != maths::graph_flavour::directed) ? nEdges += 0.5 : ++nEdges;
249  }
250  }
251  }
252 
253  check_equality(static_cast<std::size_t>(nEdges), size, "Graph size wrong");
254  }
