18  template<test_mode Mode, class PartitionedData>
19  void check_details(std::string_view description, test_logger<Mode>& logger, const PartitionedData& data, const PartitionedData& prediction)
20  {
21  check_equality(combine_messages(description, "Size different"), logger, data.size(), prediction.size());
22  if(check_equality(combine_messages(description, "Number of partitions different"), logger, data.num_partitions(), prediction.num_partitions()))
23  {
24  for(std::size_t i{}; i<prediction.num_partitions(); ++i)
25  {
26  const std::string message{combine_messages(description,"Partition " + std::to_string(i))};
27  if(check_range(combine_messages(message, "iterator (const)"), logger, data.begin_partition(i), data.end_partition(i), prediction.begin_partition(i), prediction.end_partition(i)))
28  {
29  for(int64_t j{}; j<distance(prediction.begin_partition(i), prediction.end_partition(i)); ++j)
30  {
31  check_equality(combine_messages(message,"[] (const)"), logger, data[i][j], prediction[i][j]);
32  }
33  }
34 
35  check_range(combine_messages(message, "r_iterator (const)"), logger, data.rbegin_partition(i), data.rend_partition(i), prediction.rbegin_partition(i), prediction.rend_partition(i));
36  check_range(combine_messages(message, "c_iterator"), logger, data.cbegin_partition(i), data.cend_partition(i), prediction.cbegin_partition(i), prediction.cend_partition(i));
37  check_range(combine_messages(message, "cr_iterator"), logger, data.crbegin_partition(i), data.crend_partition(i), prediction.crbegin_partition(i), prediction.crend_partition(i));
38 
39  auto& r{const_cast<PartitionedData&>(prediction)};
40  auto& d{const_cast<PartitionedData&>(data)};
41  if(check_range(combine_messages(message, "iterator"), logger, d.begin_partition(i), d.end_partition(i), r.begin_partition(i), r.end_partition(i)))
42  {
43  for(int64_t j{}; j<distance(r.begin_partition(i), r.end_partition(i)); ++j)
44  {
45  check_equality(combine_messages(message,"[]"), logger, d[i][j], r[i][j]);
46  }
47  }
48 
49  check_range(combine_messages(message, "r_iterator"), logger, d.rbegin_partition(i), d.rend_partition(i), r.rbegin_partition(i), r.rend_partition(i));
50  }
51  }
52  }

54  template<test_mode Mode, class PartitionedData, class T=typename PartitionedData::value_type>
55  void check_equivalence(std::string_view description, test_logger<Mode>& logger, const PartitionedData& data, std::initializer_list<std::initializer_list<T>> prediction)
56  {
57  const auto numElements{std::accumulate(prediction.begin(), prediction.end(), std::size_t{},
58  [](std::size_t val, std::initializer_list<T> partition) { return val += partition.size();})};
59 
60  check_equality(combine_messages(description, "Number of elements"), logger, data.size(), numElements);
61 
62  if(check_equality(combine_messages(description, "Number of partitions"), logger, data.num_partitions(), prediction.size()))
63  {
64  for(std::size_t i{}; i<prediction.size(); ++i)
65  {
66  const std::string message{combine_messages(description, "Partition " + std::to_string(i))};
67  check_range(message + ": iterator", logger, data.begin_partition(i), data.end_partition(i), (prediction.begin() + i)->begin(), (prediction.begin() + i)->end());
68 
69  check_range(message + ": riterator", logger, data.rbegin_partition(i), data.rend_partition(i), std::rbegin(*(prediction.begin() + i)), std::rend(*(prediction.begin() + i)));
70  }
71  }
72  }
