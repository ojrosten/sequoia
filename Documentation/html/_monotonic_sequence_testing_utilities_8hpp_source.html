18  template<test_mode Mode, class T>
19  void check(std::string_view description, test_logger<Mode>& logger, const T& sequence, const T& prediction)
20  {
21  if(check_equality(combine_messages(description, "Size incorrect"), logger, sequence.size(), prediction.size()))
22  {
23  if(!prediction.empty())
24  {
25  check_equality(combine_messages(description, "Back element wrong"), logger, sequence.back(), prediction.back());
26  check_equality(combine_messages(description, "Front element wrong"), logger, sequence.front(), prediction.front());
27  }
28 
29  auto i_prediction{prediction.begin()}, i{sequence.begin()};
30  auto ci_prediction{prediction.cbegin()}, ci{sequence.cbegin()};
31  auto ri_prediction{prediction.rbegin()}, ri{sequence.rbegin()};
32  auto cri_prediction{prediction.crbegin()}, cri{sequence.crbegin()};
33 
34  for(;i_prediction != prediction.end(); ++i_prediction, ++i, ++ci_prediction, ++ci, ++ri_prediction, ++ri, ++cri_prediction, ++cri)
35  {
36  using std::distance;
37  const auto d{distance(prediction.begin(), i_prediction)};
38  const std::string mess{" for index " + std::to_string(d)};
39 
40  check_equality(combine_messages(description, "Dereferenced iterator wrong" + mess), logger, *i, *i_prediction);
41  check_equality(combine_messages(description, "Dereferenced citerator wrong" + mess), logger, *ci, *ci_prediction);
42 
43  check_equality(combine_messages(description, "operator[] wrong" + mess), logger, sequence[d], prediction[d]);
44 
45  const auto shift{static_cast<int64_t>(prediction.size()) - d - 1};
46  check_equality(combine_messages(description, "Dereferenced riterator wrong" + mess), logger, *(ri + shift), *(ri_prediction + shift));
47  check_equality(combine_messages(description, "Dereferenced criterator wrong" + mess), logger, *(cri + shift), *(cri_prediction + shift));
48  }
49 
50  unit_testing::check(combine_messages(description, "iterator location wrong"), logger, i_prediction == prediction.end());
51  unit_testing::check(combine_messages(description, "citerator location wrong"), logger, ci_prediction == prediction.cend());
52  unit_testing::check(combine_messages(description, "riterator location wrong"), logger, ri_prediction == prediction.rend());
53  unit_testing::check(combine_messages(description, "criterator location wrong"), logger, cri_prediction == prediction.crend());
54  }
55  }
