68  template<bool Checked=true, class UnaryOp>
69  constexpr void mutate(const_iterator first, const_iterator last, UnaryOp op)
70  {
71  using std::distance;
72  if constexpr(Checked)
73  {
74  if(first != last)
75  {
76  auto tmp{m_Sequence};
77  auto f{tmp.begin() + distance(cbegin(), first)};
78  auto l{tmp.begin() + distance(cbegin(), last)};
79 
80  for(auto i{f}; i != l; ++i)
81  {
82  *i = op(*i);
83  }
84 
85  if(f != tmp.begin()) --f;
86  if(l != tmp.end()) ++l;
87 
88  while((f != l) && (f + 1) != tmp.end())
89  {
90  if(Compare{}(*f, *(f+1)))
91  throw std::logic_error("monotonic_sequence::mutate - monotonicity violated");
92 
93  ++f;
94  }
95 
96  m_Sequence = std::move(tmp);
97  }
98  }
99  else
100  {
101  while(first != last)
102  {
103  auto pos{m_Sequence.begin() + distance(cbegin(), first++)};
104  *pos = op(*pos);
105  }
106  }
