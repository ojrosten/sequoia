826  auto check_relative_performance(Logger& logger, F fast, S slow, const double minSpeedUp, const double maxSpeedUp, const std::size_t trials, const double num_sds, const bool reportSuccess, std::string_view description) -> performance_results<std::invoke_result_t<F>>
827  {

829  static_assert(std::is_same_v<R, std::invoke_result_t<S>>, "Fast/Slow invokables must have same return value");
830 
831  // Replace with contracts in 2020
832  if((minSpeedUp <= 1) || (maxSpeedUp <= 1))
833  throw std::logic_error("Relative performance test requires speed-up factors > 1!");
834 
835  if(minSpeedUp > maxSpeedUp)
836  throw std::logic_error("maxSpeedUp must be >= minSpeedUp");

839  r.log_performance_check();

844  using namespace maths;

846  std::vector<double> fastData, slowData;
847  fastData.reserve(trials);
848  slowData.reserve(trials);
849 
850  std::random_device generator;
851  for(std::size_t i{}; i < trials; ++i)
852  {
853  std::packaged_task<R()> fastTask{fast}, slowTask{slow};
854 
855  results.fast_futures.emplace_back(std::move(fastTask.get_future()));
856  results.slow_futures.emplace_back(std::move(slowTask.get_future()));
857

859 
860  std::uniform_real_distribution<double> distribution{0.0, 1.0};
861  const bool fastFirst{(distribution(generator) < 0.5)};
862 
863  start = steady_clock::now();
864  fastFirst ? fastTask() : slowTask();
865  end = steady_clock::now();
866 
867  duration<double> duration = end - start;
868  fastFirst ? fastData.push_back(duration.count()) : slowData.push_back(duration.count());
869 
870  start = steady_clock::now();
871  fastFirst ? slowTask() : fastTask();
872  end = steady_clock::now();
873 
874  duration = end - start;
875  fastFirst ? slowData.push_back(duration.count()) : fastData.push_back(duration.count());
876  }
877 
878  auto compute_stats{
879  [](auto first, auto last) {
880  const auto data{sample_standard_deviation(first, last)};
881  return std::make_pair(data.first.value(), data.second.value());
882  }
883  };
884 
885  const auto [sig_f, m_f]{compute_stats(fastData.cbegin(), fastData.cend())};
886  const auto [sig_s, m_s]{compute_stats(slowData.cbegin(), slowData.cend())};
887 
888  using namespace maths::bias;
889  if(m_f + sig_f < m_s - sig_s)
890  {
891  if(sig_f >= sig_s)
892  {
893  results.passed = (minSpeedUp * m_f <= (m_s + num_sds * sig_s))
894  && (maxSpeedUp * m_f >= (m_s - num_sds * sig_s));
895  }
896  else
897  {
898  results.passed = (m_s / maxSpeedUp <= (m_f + num_sds * sig_f))
899  && (m_s / minSpeedUp >= (m_f - num_sds * sig_f));
900  }
901  }
902 
903  auto serializer{
904  [m_f=m_f,sig_f=sig_f,m_s=m_s,sig_s=sig_s,num_sds,minSpeedUp,maxSpeedUp](){
905  std::ostringstream message;
906  message << "\n\tFast Task duration: " << m_f << "s";
907  message << " +- " << num_sds << " * " << sig_f;
908  message << "\n\tSlow Task duration: " << m_s << "s";
909  message << " +- " << num_sds << " * " << sig_s;
910  message << " [" << m_s / m_f << "; (" << minSpeedUp << ", " << maxSpeedUp << ")]";
911 
912  return message.str();
913  }
914  };
915 
916  if(!results.passed)
917  {
918  const std::string message{logger.failure_description(description) + serializer()};
919  logger.log_performance_failure(message);
920  }
921  else if(reportSuccess)
922  {
923  logger.post_message(serializer());
924  }
