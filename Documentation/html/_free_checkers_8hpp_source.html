162  template<test_mode Mode, class T>
163  bool dispatch_check(std::string_view description, unit_test_logger<Mode>& logger, equality_tag, const T& value, const T& prediction)
164  {
165  constexpr bool delegate{has_detailed_equality_checker_v<T> || is_container_v<T>};
166 
167  static_assert(delegate || is_equal_to_comparable_v<T>,
168  "Provide either a specialization of detailed_equality_checker or ensure operator== exists, together with a specialization of serializer");
169 
170  using sentinel = typename unit_test_logger<Mode>::sentinel;
171  sentinel s{logger, add_type_info<T>(description)};
172 
173  const auto priorFailures{logger.failures()};
174 
175  auto messageGenerator{
176  [description](std::string op, std::string retVal){
177  std::string info{add_type_info<T>("")
178  .append("\toperator").append(std::move(op))
179  .append(" returned ").append(std::move(retVal)).append("\n")
180  };
181  return description.empty() ? std::move(info) : std::string{"\t"}.append(description).append("\n" + std::move(info));
182  }
183  };
184 
185  if constexpr(is_equal_to_comparable_v<T>)
186  {
187  s.log_check();
188  if(!(prediction == value))
189  {
190  auto message{messageGenerator("==", "false")};
191  if constexpr(!delegate)
192  {
193  message.append("\tObtained : " + to_string(value) + "\n");
194  message.append("\tPredicted: " + to_string(prediction) + "\n\n");
195  }
196  logger.log_failure(message);
197  }
198  }
199 
200  if constexpr(delegate)
201  {
202  if constexpr(is_not_equal_to_comparable_v<T>)
203  {
204  s.log_check();
205  if(prediction != value)
206  {
207  logger.log_failure(messageGenerator("!=", "true"));
208  }
209  }
210 
211  if constexpr(has_detailed_equality_checker_v<T>)
212  {
213  detailed_equality_checker<T>::check(description, logger, value, prediction);
214  }
215  else if constexpr(is_container_v<T>)
216  {
217  check_range(description, logger, std::begin(value), std::end(value), std::begin(prediction), std::end(prediction));
218  }
219  else
220  {
221  static_assert(dependent_false<T>::value, "Unable to check detailed equality for Type");
222  }
223  }
224 
225  return logger.failures() == priorFailures;
226  }
