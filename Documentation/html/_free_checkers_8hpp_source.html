191  template<test_mode Mode, class T>
192  bool dispatch_check(std::string_view description, test_logger<Mode>& logger, equality_tag, const T& value, const T& prediction)
193  {
194  constexpr bool delegate{has_detailed_equality_checker_v<T> || is_container_v<T>};
195 
196  static_assert(delegate || is_equal_to_comparable_v<T>,
197  "Provide either a specialization of detailed_equality_checker or ensure operator== exists, together with a specialization of serializer");
198 
199  using sentinel = typename test_logger<Mode>::sentinel;
200  sentinel s{logger, add_type_info<T>(description)};
201 
202  const auto priorFailures{logger.failures()};
203 
204  auto messageGenerator{
205  [description](std::string op, std::string retVal){
206  std::string info{add_type_info<T>("")
207  .append("\toperator").append(std::move(op))
208  .append(" returned ").append(std::move(retVal)).append("\n")
209  };
210  return description.empty() ? std::move(info) : std::string{"\t"}.append(description).append("\n" + std::move(info));
211  }
212  };
213 
214  if constexpr(is_equal_to_comparable_v<T>)
215  {
216  s.log_check();
217  if(!(prediction == value))
218  {
219  auto message{messageGenerator("==", "false")};
220  if constexpr(!delegate)
221  {
222  message.append("\tObtained : " + to_string(value) + "\n");
223  message.append("\tPredicted: " + to_string(prediction) + "\n\n");
224  }
225  logger.log_failure(message);
226  }
227  }
228 
229  if constexpr(delegate)
230  {
231  if constexpr(is_not_equal_to_comparable_v<T>)
232  {
233  s.log_check();
234  if(prediction != value)
235  {
236  logger.log_failure(messageGenerator("!=", "true"));
237  }
238  }
239 
240  if constexpr(has_detailed_equality_checker_v<T>)
241  {
242  detailed_equality_checker<T>::check(description, logger, value, prediction);
243  }
244  else if constexpr(is_container_v<T>)
245  {
246  check_range(description, logger, std::begin(value), std::end(value), std::begin(prediction), std::end(prediction));
247  }
248  else
249  {
250  static_assert(dependent_false<T>::value, "Unable to check detailed equality for Type");
251  }
252  }
253 
254  return logger.failures() == priorFailures;
255  }

308  bool check_range(std::string_view description, test_logger<Mode>& logger, ElementDispatchDiscriminator discriminator, Iter first, Iter last, PredictionIter predictionFirst, PredictionIter predictionLast)
309  {
310  typename test_logger<Mode>::sentinel r{logger, description};
311  bool equal{true};
312 
313  using std::distance;
314  using std::advance;
315 
316  const auto predictedSize{distance(predictionFirst, predictionLast)};
317  if(check_equality(combine_messages(description, "Container size wrong", "\n"), logger, distance(first, last), predictedSize))
318  {
319  auto predictionIter{predictionFirst};
320  auto iter{first};
321  for(; predictionIter != predictionLast; advance(predictionIter, 1), advance(iter, 1))
322  {
323  std::string dist{std::to_string(distance(predictionFirst, predictionIter)).append("\n")};
324  if(!dispatch_check(combine_messages(description, "element ", "\n").append(std::move(dist)), logger, discriminator, *iter, *predictionIter)) equal = false;
325  }
326  }
327  else
328  {
329  equal = false;
330  }
331 
332  return equal;
333  }
