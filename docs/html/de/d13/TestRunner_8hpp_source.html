280  void add_test_family(std::string_view name, Test&& test, Tests&&... tests)
281  {
282  const bool done{
283  [this, name](Test&& test, Tests&&... tests){
284  if(!m_SelectedSources.empty())
285  {
286  test_family f{name, m_Paths.tests(), m_Paths.test_materials(), m_Paths.output(), m_Recovery};
287  add_tests(f, std::forward<Test>(test), std::forward<Tests>(tests)...);
288  if(!f.empty())
289  {
290  m_Families.push_back(std::move(f));
291  mark_family(name);
292  return true;
293  }
294  }
295 
296  return false;
297  }(std::forward<Test>(test), std::forward<Tests>(tests)...)
298  };
299 
300  if(!done && (m_SelectedSources.empty() || !m_SelectedFamilies.empty()))
301  {
302  if(mark_family(name))
303  {
304  m_Families.emplace_back(name, m_Paths.tests(), m_Paths.test_materials(), m_Paths.output(), m_Recovery,
305  std::forward<Test>(test), std::forward<Tests>(tests)...);
306  }
307  }
