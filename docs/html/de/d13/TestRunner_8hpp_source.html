350  {
351  const bool done{

353  if(!m_SelectedSources.empty())
354  {
355  test_family f{name, m_Paths.tests(), m_Paths.test_materials(), m_Paths.output(), m_Recovery};
356  add_tests(f, std::forward<Test>(test), std::forward<Tests>(tests)...);
357  if(!f.empty())

360  mark_family(name);
361  return true;

365  return false;

367  };
