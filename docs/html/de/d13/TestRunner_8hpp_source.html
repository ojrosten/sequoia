350 template<class Filter, concrete_test... Tests>
351 requires (sizeof...(Tests) > 0)
352 void extract_suite_tree(std::string_view name, Filter&& filter, Tests&&... tests)
353 {
354 using namespace object;
355 using namespace maths;
356
357 if(!m_Suites.order())
358 {
359 m_Suites.add_node(suite_type::npos, log_summary{});
360 }
361
362 std::vector<std::filesystem::path> materialsPaths{};
363
364 extract_tree(suite{std::string{name}, std::forward<Tests>(tests)...},
365 std::forward<Filter>(filter),
366 overloaded{
367 [] <class... Ts> (const suite<Ts...>&s) -> suite_node { return {.summary{log_summary{s.name}}}; },
368 [this, name, &materialsPaths]<concrete_test T>(T&& test) -> suite_node {
369 test.initialize(name,
370 test.source_file(),
371 proj_paths(),
372 set_materials(test.source_file(), proj_paths(), materialsPaths),
373 make_active_recovery_paths(m_RecoveryMode, proj_paths()));
374
375 return {.summary{log_summary{test.name()}}, .optTest{std::move(test)}};
376 }
377 },
378 m_Suites,
379 0);
380 }

382 template<concrete_test... Tests>
383 requires (sizeof...(Tests) > 0)
384 static void check_for_duplicates(std::string_view name, const Tests&... tests)
385 {
386 using duplicate_set = std::set<std::pair<std::string_view, std::filesystem::path>>;
387
388 duplicate_set namesAndSources{};
389
390 auto check{
391 [&,name](concrete_test auto const& test) {
392 if(!namesAndSources.emplace(test.name(), test.source_file()).second)
393 throw std::runtime_error{duplication_message(name, test.name(), test.source_file())};
394 }
395 };
396
397 (check(tests), ...);
398 }
