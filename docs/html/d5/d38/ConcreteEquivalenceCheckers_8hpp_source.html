133  template<test_mode Mode>
134  static void check(test_logger<Mode>& logger, const std::filesystem::path& path, const std::filesystem::path& prediction)
135  {
136  namespace fs = std::filesystem;
137 
138  const auto pathType{fs::status(path).type()};
139  const auto predictionType{fs::status(prediction).type()};
140 
141  if(check_equality(preamble("Path type", path, prediction), logger, pathType, predictionType))
142  {
143  if(!path.empty())
144  {
145  const auto pathFinalToken{*(--path.end())};
146  const auto predictionFinalToken{*(--prediction.end())};
147  if(check_equality("Final path token", logger, pathFinalToken, predictionFinalToken))
148  {
149  switch(pathType)
150  {
151  case fs::file_type::regular:
152  check_file(logger, path, prediction);
153  break;
154  case fs::file_type::directory:

157  default:
158  throw std::logic_error{std::string{"Detailed equivalance check for paths of type "}
159  .append(serializer<fs::file_type>::make(pathType)).append(" not currently implemented")};

161  }
162  }

167  template<test_mode Mode>
168  static void check_directory(test_logger<Mode>& logger, const std::filesystem::path& dir, const std::filesystem::path& prediction)
169  {
170  namespace fs = std::filesystem;
171 
172  auto generator{
173  [](const fs::path& dir){
174  std::vector<fs::path> paths{};
175  for(auto& p : fs::directory_iterator(dir))
176  {
177  if(p.path().filename() != ".DS_Store") paths.push_back(p);
178  }
179 
180  std::sort(paths.begin(), paths.end());
181 
182  return paths;
183  }
184  };
185 
186  const std::vector<fs::path> paths{generator(dir)}, predictedPaths{generator(prediction)};
187 
188  check_equality(std::string{"Number of directory entries for "}.append(dir), logger, paths.size(), predictedPaths.size());
189 
190  const auto iters{std::mismatch(paths.begin(), paths.end(), predictedPaths.begin(), predictedPaths.end(),
191  [&dir,&prediction](const fs::path& lhs, const fs::path& rhs){
192  return fs::relative(lhs, dir) == fs::relative(rhs, prediction);
193  })};
194  if((iters.first != paths.end()) && (iters.second != predictedPaths.end()))
195  {
196  check_equality("First directory entry mismatch", logger, *iters.first, *iters.second);
197  }
198  else if(iters.first != paths.end())
199  {
200  check_equality("First directory entry mismatch", logger, *iters.first, fs::path{});
201  }
202  else if(iters.second != predictedPaths.end())
203  {
204  check_equality("First directory entry mismatch", logger, fs::path{}, *iters.second);
205  }
206  else
207  {
208  for(std::size_t i{}; i < paths.size(); ++i)
209  {
210  check(logger, paths[i], predictedPaths[i]);
211  }
212  }
