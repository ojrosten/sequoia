55  template<class... Args>
56  requires (constructible_from<Products, Args...> && ...)
57  factory(const std::array<string_view, size()>& names, const Args&... args)
58  : m_Creators{make_array(names, std::make_index_sequence<size()>{}, args...)}
59  {
60  std::sort(m_Creators.begin(), m_Creators.end(), [](const element& lhs, const element& rhs){ return lhs.first < rhs.first; });
61 
62  auto comp{[](const element& lhs, const element& rhs) { return lhs.first == rhs.first; }};
63  if(std::adjacent_find(m_Creators.cbegin(), m_Creators.cend(), comp) != m_Creators.cend())
64  throw std::logic_error{"Factory product names must be unique!"};
65  }

67  [[nodiscard]]
68  vessel create(std::string_view name) const
69  {
70  auto found{std::lower_bound(m_Creators.begin(), m_Creators.end(), name,
71  [](const element& e, std::string_view n) { return e.first < n; } )};
72 
73  if((found == m_Creators.end()) || (found->first != name))
74  throw std::runtime_error{std::string{"Factory unable to create product of name '"}.append(name).append("'")};
75

77  };
