100 factory(const std::array<std::string_view, size()>& names)
101 : m_Creators{make_array(names, std::make_index_sequence<size()>{})}
102 {
103 std::sort(m_Creators.begin(), m_Creators.end(), [](const element& lhs, const element& rhs){ return lhs.first < rhs.first; });
104
105 auto comp{[](const element& lhs, const element& rhs) { return lhs.first == rhs.first; }};
106 if(std::adjacent_find(m_Creators.cbegin(), m_Creators.cend(), comp) != m_Creators.cend())
107 throw std::logic_error{"Factory product names must be unique!"};

117 if(found == m_Creators.end())

120 return std::visit(overloaded{[&](const auto& v) { return vessel{v.make(std::forward<Args>(args)...)}; }}, found->second);

127 vessel make_or(std::string_view name, Args&&... args) const
128 {
129 auto found{find(name)};
130 if(found == m_Creators.end())
131 {
132 found = std::find_if(m_Creators.begin(), m_Creators.end(),
133 [](const element& e){ return std::holds_alternative<product_creator<Product>>(e.second); });
