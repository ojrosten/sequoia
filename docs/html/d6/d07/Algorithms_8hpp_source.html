26  constexpr void bubble_up(Iter begin, Iter current, Comparer comp = Comparer{})
27  {
28  if(current == begin) return;
29 
30  using namespace std;
31  auto parent{begin + (distance(begin, current) - 1) / 2};
32  if(comp(*parent, *current))
33  {

35  iter_swap(parent, current);

40  template<class Iter, class Comparer=std::less<std::decay_t<decltype(*Iter())>>>
41  constexpr void bubble_down(Iter begin, Iter current, Iter end, Comparer comp = Comparer{})
42  {
43  using namespace std;
44  if(distance(begin, end) <= 1) return;
45 
46  if(2*(distance(begin, current) + 1) < distance(begin, end))
47  {
48  auto rightChild{begin + 2*(distance(begin, current) + 1)};
49  auto leftChild{rightChild - 1};
50 
51  auto dominantChild{comp(*rightChild, *leftChild) ? leftChild : rightChild};
52  if(comp(*current, *dominantChild))
53  {
54  using std::iter_swap;
55  iter_swap(dominantChild, current);
56  bubble_down(begin, dominantChild, end, comp);
57  }
58  }
59  else if(2*distance(begin, current) + 1 < distance(begin, end))
60  {
61  using std::iter_swap;
62  auto leftChild{begin + 2*distance(begin, current) + 1};
63  if(comp(*current, *leftChild)) iter_swap(leftChild, current);
64  }
65  }
