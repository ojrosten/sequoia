112  constexpr ForwardIt rotate(ForwardIt first, ForwardIt n_first, ForwardIt last)
113  {
114  if(first == n_first) return last;
115  if(last == n_first) return first;

118

120  const auto distFromBegin{distance(first, n_first)};
121  const auto retIter{next(first, distToEnd)};

123  const auto dist{min(distToEnd, distFromBegin)};

126 
127  while(first != unswapped)
128  {
129  iter_swap(first++, n_first++);
130  }
131

133  sequoia::rotate(unswapped, next(unswapped, dist), last);
