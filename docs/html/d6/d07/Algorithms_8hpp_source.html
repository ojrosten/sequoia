25 template<class Iter, class Comparer=std::less<std::decay_t<decltype(*Iter())>>>
26 constexpr void bubble_up(Iter begin, Iter current, Comparer comp = Comparer{})
27 {
28 if(current == begin) return;
29
30 using namespace std;
31 auto parent{begin + (distance(begin, current) - 1) / 2};
32 if(comp(*parent, *current))
33 {
34 using std::iter_swap;
35 iter_swap(parent, current);
36 bubble_up(begin, parent, comp);
37 }
38 }

40 template<class Iter, class Comparer=std::less<std::decay_t<decltype(*Iter())>>>
41 constexpr void bubble_down(Iter begin, Iter current, Iter end, Comparer comp = Comparer{})
42 {
43 using namespace std;
44 if(distance(begin, end) <= 1) return;
45
46 if(2*(distance(begin, current) + 1) < distance(begin, end))
47 {
48 auto rightChild{begin + 2*(distance(begin, current) + 1)};
49 auto leftChild{rightChild - 1};
50
51 auto dominantChild{comp(*rightChild, *leftChild) ? leftChild : rightChild};
52 if(comp(*current, *dominantChild))
53 {
54 using std::iter_swap;
55 iter_swap(dominantChild, current);
56 bubble_down(begin, dominantChild, end, comp);
57 }
58 }
59 else if(2*distance(begin, current) + 1 < distance(begin, end))
60 {
61 using std::iter_swap;
62 auto leftChild{begin + 2*distance(begin, current) + 1};
63 if(comp(*current, *leftChild)) iter_swap(leftChild, current);
64 }
65 }

67 template<class Iter, class Comparer=std::less<std::decay_t<decltype(*Iter())>>>
68 constexpr void make_heap(Iter begin, Iter end, Comparer comp = Comparer{})
69 {
70 using namespace std;
71 if(distance(begin, end) <= 1) return;
72
73 auto current{begin+1};
74 while(current != end)
75 {
76 bubble_up(begin, current, comp);
77 ++current;
78 }

81 // One reason to retain this is that it uses iter_swap internally.
82 // This means that the behaviour can be customized by specializing
83 // the latter. Currently, this is exploited to sort graph nodes.
84 template<class Iter, class Comparer=std::less<std::decay_t<decltype(*Iter())>>>
85 constexpr void sort(Iter begin, Iter end, Comparer comp = Comparer{})
86 {
87 using std::distance;
88 if(distance(begin, end) <= 1) return;
89
90 sequoia::make_heap(begin, end, comp);
91 while(end != begin)
92 {
93 using std::iter_swap;
94 iter_swap(begin, end-1);
95 bubble_down(begin, begin, end-1, comp);
96 --end;
97 }
98 }

129 template<std::forward_iterator Iter, class Comparer=std::equal_to<std::decay_t<decltype(*Iter())>>>
130 constexpr void cluster(Iter begin, Iter end, Comparer comp = Comparer{})
131 {
132 if(begin == end) return;
133
134 auto current{begin};
135 while((current != end) && comp(*current, *begin)) ++current;
136
137 auto endOfCluster{current};
138
139 while(current != end)

141 if(comp(*current, *begin))

143 using std::iter_swap;

146 }
147
148 ++current;
149 }
150

152 }
