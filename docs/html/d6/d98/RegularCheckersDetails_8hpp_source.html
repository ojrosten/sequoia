31 template<test_mode Mode, class Actions, pseudoregular T, class... Args>
32 bool do_check_copy_assign(test_logger<Mode>& logger, [[maybe_unused]] const Actions& actions, T& z, const T& y, const Args&... args)
33 {
34 z = y;

56 template<test_mode Mode, class Actions, pseudoregular T, std::invocable<T&> Mutator, class... Args>
57 bool check_semantics(test_logger<Mode>& logger, const Actions& actions, const T& x, const T& y, Mutator yMutator, const Args&... args)
58 {
59 sentinel<Mode> sentry{logger, ""};
60
61 if(!check_preconditions(logger, actions, x, y, args...))
62 return false;
63
64 T z{x};
65 const bool consistentCopy{
66 check(equality, "Inconsistent copy constructor (x)", logger, z, x)
67 };
68
69 if constexpr(has_post_copy_action<Actions, test_logger<Mode>, T, T, Args...>)
70 {
71 if(consistentCopy)
72 {
73 actions.post_copy_action(logger, z, T{y}, args...);
74 }
75 }
76
77 const bool consistentCopyAssign{check_copy_assign(logger, actions, z, y, args...)};
78 // z == y, if copy assign is consistent, even if copy construction is not
79
80 if(consistentCopyAssign)
81 {
82 check_move_construction(logger, actions, std::move(z), y, opt_moved_from_ref<T>{}, args...);
83 }
84
85 if(!consistentCopy)
86 return false;
87
88 T w{x};
89 check_move_assign(logger, actions, w, T{y}, y, opt_moved_from_ref<T>{}, yMutator, args...);
90
91 if constexpr (do_swap<Args...>::value)
92 {
93 if (consistentCopy)
94 {
95 check_swap(logger, actions, T{x}, T{y}, x, y, yMutator, args...);
96 }
97 }
98
99 if constexpr(!std::is_same_v<std::remove_cvref_t<Mutator>, null_mutator>)
100 {
101 if(consistentCopy && consistentCopyAssign)
102 {
103 T v{y};
104 yMutator(v);
105
106 if(check(
107 "Either mutation is not doing anything following copy construction"
108 " or value semantics are broken, with mutation of an object also changing"
109 " the object from which it was copied", logger, v != y))
110 {
111 v = y;
112 if(check(equality, "Inconsistent copy assignment (from mutated y)", logger, v, y))
113 {
114 yMutator(v);
115
116 check(
117 "Either mutation is not doing anything following copy assignment"
118 " or value semantics are broken, with mutation of an object also changing"
119 " the object from which it was assigned", logger, v != y);
120 }
121 }
122 }
123 }
124
125 if constexpr (serializable_to<T, std::stringstream> && deserializable_from<T, std::stringstream>)
126 {
127 check_serialization(logger, actions, T{x}, y, args...);
128 }
129
130 return !sentry.failure_detected();
131 }
