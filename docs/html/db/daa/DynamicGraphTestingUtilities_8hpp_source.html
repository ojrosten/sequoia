168 void creation_permutations()
169 {
170 using namespace object;
171 using EW = EdgeWeight;
172 using NW = NodeWeight;
173 using ESTraits = EdgeStorageTraits;
174 using NSTraits = NodeWeightStorageTraits;
175
176 run_tests<GraphFlavour, faithful_producer<EW>, faithful_producer<NW>, ESTraits, NSTraits>();
177
178 if constexpr(!minimal_graph_tests())
179 {
180 if constexpr(!std::is_empty_v<NodeWeight>)
181 {
182 run_tests<GraphFlavour, faithful_producer<EW>, data_pool<NW>, ESTraits, NSTraits>();
183 }
184
185 if constexpr(!std::is_empty_v<EdgeWeight>)
186 {
187 run_tests<GraphFlavour, data_pool<EW>, faithful_producer<NW>, ESTraits, NSTraits>();
188 }
189
190 if constexpr(!std::is_empty_v<EdgeWeight> && !std::is_empty_v<NodeWeight>)
191 {
192 run_tests<GraphFlavour, data_pool<EW>, data_pool<NW>, ESTraits, NSTraits>();
193 }
194 }
