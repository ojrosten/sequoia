219  [[nodiscard]]
220  family_results execute(update_mode updateMode,
221  concurrency_mode concurrenyMode,
222  std::optional<std::size_t> index)
223  {
224  family_processor processor{updateMode};
225  auto pathsMaker{
226  [&info=m_Info](auto& test) -> paths {
227  return {test.source_filename(),
228  test.working_materials(),
229  test.predictive_materials(),
230  info.output_dir(),
231  info.test_repo()};
232  }
233  };
234 
235  if(concurrenyMode < concurrency_mode::unit)
236  {
237  auto process{
238  [&processor,pathsMaker,index](auto& optTest) {
239  if(optTest.has_value())
240  {
241  processor.process(optTest->execute(index), pathsMaker(*optTest));
242  }
243  }
244  };
245 
246  std::apply([process](auto&... optTest) { (process(optTest), ...); }, m_Tests);
247  }
248  else
249  {
250  using data = std::pair<log_summary, paths>;
251  std::vector<std::future<data>> results{};
252 
253  auto generator{
254  [&results,pathsMaker,index](auto& optTest) {
255  if(optTest.has_value())
256  {
257  results.emplace_back(
258  std::async([&test=*optTest,pathsMaker,index](){
259  return std::make_pair(test.execute(index), pathsMaker(test)); })
260  );
261  }
262  }
263  };
264 
265  std::apply([generator](auto&... optTest) { (generator(optTest), ...); }, m_Tests);
266 
267  for(auto& r : results)
268  {
269  const auto[summary, paths]{r.get()};
270  processor.process(summary, paths);
271  }
272  }
273 
274  return processor.finalize_and_acquire();
275  }
