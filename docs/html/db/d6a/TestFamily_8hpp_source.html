189 void add_test(std::vector<std::filesystem::path>& materialsPaths, T test)

191 using test_type = std::optional<std::remove_cvref_t<T>>;

198 [[nodiscard]]
199 family_results execute(update_mode updateMode,
200 concurrency_mode concurrenyMode,
201 std::optional<std::size_t> index)
202 {
203 family_processor processor{updateMode};
204 auto pathsMaker{
205 [&info=m_Info](auto& test) -> test_paths {
206 return {test.source_filename(),
207 test.working_materials(),
208 test.predictive_materials(),
209 info.proj_paths()};
210 }
211 };
212
213 if(concurrenyMode < concurrency_mode::unit)
214 {
215 auto process{
216 [&processor,pathsMaker,index](auto& optTest) {
217 if(optTest.has_value())
218 {
219 processor.process(optTest->execute(index), pathsMaker(*optTest));
220 }
221 }
222 };
223
224 std::apply([process](auto&... optTest) { (process(optTest), ...); }, m_Tests);
225 }
226 else
227 {
228 using data = std::pair<log_summary, test_paths>;
229 std::vector<std::future<data>> results{};
230
231 auto generator{
232 [&results,pathsMaker,index](auto& optTest) {
233 if(optTest.has_value())
234 {
235 results.emplace_back(
236 std::async([&test=*optTest,pathsMaker,index](){
237 return std::make_pair(test.execute(index), pathsMaker(test)); })
238 );
239 }
240 }
241 };
242
243 std::apply([generator](auto&... optTest) { (generator(optTest), ...); }, m_Tests);
244
245 for(auto& r : results)
246 {
247 const auto[summary, test_paths]{r.get()};
248 processor.process(summary, test_paths);
249 }
250 }
251
252 return processor.finalize_and_acquire();
253 }

277 {

279
280 auto reset{
281 [this,&materialsPaths](auto& optTest){
282 if(optTest)
283 {
284 using type = typename std::remove_cvref_t<decltype(optTest)>::value_type;
