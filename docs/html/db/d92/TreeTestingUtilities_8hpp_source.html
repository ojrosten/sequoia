47 template<class CheckType, test_mode Mode>
48 static size_type check_node(CheckType flavour, test_logger<Mode>& logger, size_type node, size_type parent, const tree_type& actual, const maths::tree_initializer<NodeWeight>& prediction)
49 {
50 if (node == tree_type::npos) return node;
51
52 if (testing::check("Insufficient nodes detected while checking node " + std::to_string(node), logger, node < actual.order()))
53 {
54 if constexpr (TreeLinkDir != maths::tree_link_direction::backward)
55 {
56 check(flavour, "Node weight for node " + std::to_string(node), logger, actual.cbegin_node_weights()[node], prediction.node);
57
58 if (auto optIter{ check_num_edges(logger, node, parent, actual, prediction) })
59 {
60 for (const auto& child : prediction.children)
61 {
62 if (!check_edge(logger, (*optIter)++, ++node, actual))
63 return tree_type::npos;
64
65 node = check_node(flavour, logger, node, optIter->partition_index(), actual, child);
66
67 if (node == tree_type::npos) return tree_type::npos;
68 }
69
70 return node;
71 }
72 }
73 else
74 {
75 if (auto optIter{ check_num_edges(logger, node, parent, actual, prediction) })
76 {
77 check(flavour, "Node weight for node " + std::to_string(node), logger, actual.cbegin_node_weights()[node], prediction.node);
78
79 for (const auto& child : prediction.children)
80 {
81 node = check_node(flavour, logger, ++node, optIter->partition_index(), actual, child);
82
83 if (node == tree_type::npos) return tree_type::npos;
84 }
85
86 if (const auto next{ node + 1 }; next < actual.order())
87 {
88 if (auto begin{ actual.cbegin_edges(next) }; begin != actual.cend_edges(next))
89 {
90 if (!testing::check("Extraneous nodes joined to node " + std::to_string(node), logger, begin->target_node() != optIter->partition_index()))
91 {
92 return tree_type::npos;
93 }
94 }
95 }
96
97 return node;
98 }
99 }
100 }
101
102 return tree_type::npos;
103 }

105 template<test_mode Mode>
106 static std::optional<edge_iterator> check_num_edges(test_logger<Mode>& logger, size_type node, [[maybe_unused]] size_type parent, const tree_type& actual, const maths::tree_initializer<NodeWeight>& prediction)
107 {
108 using std::distance;
109
110 if constexpr (TreeLinkDir == maths::tree_link_direction::forward)
111 {
112 if (check(equality, "Number of children for node " + std::to_string(node), logger, static_cast<std::size_t>(distance(actual.cbegin_edges(node), actual.cend_edges(node))), prediction.children.size()))
113 return actual.cbegin_edges(node);
114 }
115 else
116 {
117 const auto begin{ actual.cbegin_edges(node) };
118 const auto dist{ static_cast<std::size_t>(distance(begin, actual.cend_edges(node))) };
119 using dist_t = decltype(dist);
120
121 const auto num{
122 [&logger,parent,dist,begin]() -> std::optional<dist_t> {
123 if (parent == tree_type::npos) return dist;
124
125 if (testing::check("Return edge detected", logger, dist > 0)
126 && check(equality, "Return edge target", logger, begin->target_node(), parent))
127 return dist - 1;
128
129 return std::nullopt;
130 }()
131 };
132
133 if (num.has_value())
134 {
135 if constexpr (TreeLinkDir == maths::tree_link_direction::symmetric)
136 {
137 if (check(equality, "Number of children for node " + std::to_string(node), logger, num.value(), prediction.children.size()))
138 return num < dist ? std::next(begin) : begin;
139 }
140 else
141 {
142 if (check(equality, "No reachable children for node " + std::to_string(node), logger, num.value(), size_type{}))
143 return num < dist ? std::next(begin) : begin;
144 }
145 }
146 }
147
148 return std::nullopt;
149 }

153 {

156 if constexpr (TreeLinkDir == maths::tree_link_direction::forward)
157 {

159 const auto dist{ distance(actual.cbegin_edges(parent), iter) };

161
162 return check(equality, mess, logger, iter->target_node(), nodeCounter);
163 }
164 else if constexpr (TreeLinkDir == maths::tree_link_direction::backward)
165 {
166 static_assert(dependent_false<tree_type>::value, "Not yet implemented");
