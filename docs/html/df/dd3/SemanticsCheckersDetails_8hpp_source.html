235  template<test_mode Mode, class Actions, orderable T, class... Args>
236  [[nodiscard]]
237  bool check_ordering_consistency(test_logger<Mode>& logger, const Actions& actions, const T& x, const T& y, const Args&... args)
238  {
239  if(!check_ordering_operators(logger, actions, x, y, args...)) return false;
240 
241  auto comp{
242  [&logger](const T& x, const T& y){
243  sentinel sentry{logger, ""};
244 
245  check("operator> and operator< are inconsistent", logger, y > x);
246  check("operator< and operator<= are inconsistent", logger, x <= y);
247  check("operator< and operator>= are inconsistent", logger, y >= x);

255  }
256  };
