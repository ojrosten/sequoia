768 template<std::floating_point T, T Lower, T Upper=std::numeric_limits<T>::infinity()>
769 requires (Upper > Lower)
770 struct interval_validator
771 {
772 constexpr static T lower{Lower}, upper{Upper};
773
774 template<std::floating_point U>
775 requires (sizeof(U) <= sizeof(T))
776 constexpr U operator()(const U val) const
777 {
778 if constexpr(Lower > -std::numeric_limits<T>::infinity())
779 {
780 if(const auto ulower{static_cast<U>(Lower)}; val < ulower)
781 throw std::domain_error{std::format("interval_validator: invoked with {}, but values should be >= {} ", val, ulower)};
782 }
783
784 if constexpr(Upper < std::numeric_limits<T>::infinity())
785 {
786 if(const auto uUpper{static_cast<U>(Upper)}; val > uUpper)
787 throw std::domain_error{std::format("interval_validator: invoked with {}, but values should be <= {} ", val, uUpper)};
788 }
789
790 return val;
791 };
792 };
