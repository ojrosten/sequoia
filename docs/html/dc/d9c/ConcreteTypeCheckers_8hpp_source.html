82 static void appender(std::string& mess, string_view_type sv, size_type pos, size_type count)
83 {
84 if constexpr(sizeof(char) >= sizeof(Char))
85 {
86 mess.append(sv.substr(pos, count));
87 }
88 else
89 {
90 if(pos > sv.size()) throw std::out_of_range{"pos out of range"};
91 const auto end{count > sv.size() - pos ? sv.size() : pos + count};
92 std::transform(sv.begin() + pos, sv.begin() + end, std::back_inserter(mess), [](Char c) { return static_cast<char>(c); });
93 }

96 template<class Advisor>
97 static auto make_advisor(std::string_view info, string_view_type obtained, string_view_type prediction, size_type pos, const tutor<Advisor>& advisor)
98 {
99 constexpr size_type defaultOffset{30}, defaultCount{60}, npos{string_view_type::npos};
100 const auto sz{std::min(obtained.size(), prediction.size())};
101
102 auto newlineBackwards{ [pos](string_view_type sv){ return pos < sv.size() ? sv.rfind('\n', pos) : npos; } };
103
104 const auto loc{std::min(newlineBackwards(prediction), newlineBackwards(obtained))};
105
106 const size_type offset{loc < npos ? std::min(defaultOffset, pos - loc) : defaultOffset};
107
108 const auto lpos{pos < offset ? 0 :
109 pos < sz ? pos - offset : sz - std::min(sz, offset)};
110
111 struct message{ std::string mess; bool trunc{}; };
112
113 auto make{
114 [](string_view_type sv, size_type lpos) -> message {
115 std::string mess{lpos > 0 ? "..." : ""};
116
117 const bool newline{(lpos < sv.size()) && (sv[lpos] == '\n')};
118 if(newline) mess.append("\\n");
119
120 const auto rpos{sv.find('\n', lpos+1)};
121 const auto count{rpos == npos ? defaultCount : rpos - lpos};
122
123 if(newline && (count == 1))
124 {
125 mess.append("\\n");
126 }
127 else
128 {
129 if(newline) ++lpos;
130 appender(mess, sv, lpos, count);
131 }
132
133 const bool trunc{lpos + count < sv.size()};
134 if(trunc) mess.append("...");
135
136 return {mess, trunc};
137 }
138 };
139
140 const auto[obMess,obTrunc]{make(obtained, lpos)};
141 const auto[prMess,prTrunc]{make(prediction, lpos)};
142
143 const bool trunc{lpos > 0 || obTrunc || prTrunc};
144 const auto message{append_lines(info, trunc ? "Surrounding substring(s):" : "Full strings:",
145 prediction_message(obMess, prMess))};
146
147 if constexpr(std::invocable<tutor<Advisor>, Char, Char>)
148 {
149
150 return tutor{
151 [message, advisor] (Char a, Char b) {
152 auto m{message};
153 return append_advice(m, {advisor, a, b});
154 },
155 "\n"
156 };
157 }
158 else
159 {
160 return tutor{
161 [message](const auto&, const auto&) { return message; },
162 "\n"
163 };
164 }
165 }

167 public:
168 template<test_mode Mode, class Advisor>
169 static void test(equality_check_t, test_logger<Mode>& logger, string_view_type obtained, string_view_type prediction, const tutor<Advisor>& advisor)
170 {
171 auto iters{std::mismatch(obtained.begin(), obtained.end(), prediction.begin(), prediction.end())};
172
173 if((iters.first != obtained.end()) && (iters.second != prediction.end()))
174 {
175 const auto dist{std::distance(obtained.begin(), iters.first)};
176 auto adv{make_advisor("", obtained, prediction, dist, advisor)};
177
178 const auto numLines{std::count(prediction.begin(), iters.second, '\n')};
179
180 const auto mess{
181 [dist,numLines]() {
182 std::string m{"First difference detected "};
183 numLines > 0 ? m.append("on line ").append(std::to_string(numLines+1))
184 : m.append("at character ").append(std::to_string(dist));
185
186 return m.append(":");
187 }()
188 };
189
190 check(equality, mess, logger, *(iters.first), *(iters.second), adv);
191 }
192 else if((iters.first != obtained.end()) || (iters.second != prediction.end()))
193 {
194 auto checker{
195 [&logger, obtained, prediction, &advisor](auto begin, auto iter, std::string_view state, std::string_view adjective){
196 const auto dist{std::distance(begin, iter)};
197 const auto info{std::string{"First "}.append(state).append(" character: ").append(display_character(*iter))};
198 auto adv{make_advisor(info, obtained, prediction, dist, advisor)};
199
200 const auto mess{append_lines("Lengths differ", std::string{"Obtained string is too "}.append(adjective))};
201
202 check(equality, mess, logger, obtained.size(), prediction.size(), adv);
203 }
204 };
205
206 if(iters.second != prediction.end())
207 {
208 checker(prediction.begin(), iters.second, "missing", "short");
209 }
210 else if(iters.first != obtained.end())
211 {
212 checker(obtained.begin(), iters.first, "excess", "long");
213 }
214 }

454 template<test_mode Mode, class Customization, invocable_r<bool, std::filesystem::path, std::filesystem::path> FinalTokenComparison>
455 static void check_path(test_logger<Mode>& logger, const Customization& custom, const std::filesystem::path& path, const std::filesystem::path& prediction, FinalTokenComparison compare)
456 {
457 namespace fs = std::filesystem;
458
459 const auto pathType{fs::status(path).type()};
460 const auto predictionType{fs::status(prediction).type()};
461
462 if(check(equality, path_check_preamble("Path type", path, prediction), logger, pathType, predictionType))
463 {
464 if(!path.empty())
465 {
466 const auto pathFinalToken{back(path)};
467 const auto predictionFinalToken{back(prediction)};
468 if(compare(pathFinalToken, predictionFinalToken))
469 {
470 switch(pathType)
471 {
472 case fs::file_type::regular:
473 check_file(logger, custom, path, prediction);
474 break;
475 case fs::file_type::directory:
476 check_directory(logger, custom, path, prediction, compare);
477 break;
478 default:
479 throw std::logic_error{std::string{"Detailed equivalance check for paths of type '"}
480 .append(serializer<fs::file_type>::make(pathType)).append("' not currently implemented")};
481 }
482 }
483 }
484 }

487 template<test_mode Mode, class Customization, invocable_r<bool, std::filesystem::path, std::filesystem::path> FinalTokenComparison>
488 static void check_directory(test_logger<Mode>& logger, const Customization& custom, const std::filesystem::path& dir, const std::filesystem::path& prediction, FinalTokenComparison compare)
489 {
490 namespace fs = std::filesystem;
491
492 auto generator{
493 [](const fs::path& dir) {
494 std::vector<fs::path> paths{};
495 for(const auto& p : fs::directory_iterator(dir))
496 {
497 if( std::find(excluded_files.begin(), excluded_files.end(), p.path().filename()) == excluded_files.end()
498 && std::find(excluded_extensions.begin(), excluded_extensions.end(), p.path().extension()) == excluded_extensions.end())
499 {
500 paths.push_back(p);
501 }
502 }
503
504 std::sort(paths.begin(), paths.end());
505
506 return paths;
507 }
508 };
509
510 const std::vector<fs::path> paths{generator(dir)}, predictedPaths{generator(prediction)};
511
512 check(equality, std::string{"Number of directory entries for "}.append(dir.generic_string()),
513 logger,
514 paths.size(),
515 predictedPaths.size());
516
517 const auto iters{std::mismatch(paths.begin(), paths.end(), predictedPaths.begin(), predictedPaths.end(),
518 [&dir,&prediction](const fs::path& lhs, const fs::path& rhs) {
519 return fs::relative(lhs, dir) == fs::relative(rhs, prediction);
520 })};
521 if((iters.first != paths.end()) && (iters.second != predictedPaths.end()))
522 {
523 check(equality, "First directory entry mismatch", logger, *iters.first, *iters.second);
524 }
525 else if(iters.first != paths.end())
526 {
527 check(equality, "First directory entry mismatch", logger, *iters.first, fs::path{});
528 }
529 else if(iters.second != predictedPaths.end())
530 {
531 check(equality, "First directory entry mismatch", logger, fs::path{}, *iters.second);
532 }
533 else
534 {
535 for(std::size_t i{}; i < paths.size(); ++i)
536 {
537 check_path(logger, custom, paths[i], predictedPaths[i], compare);
538 }
539 }
540 }
