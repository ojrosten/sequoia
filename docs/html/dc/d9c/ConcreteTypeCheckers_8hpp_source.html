56 static void appender(std::string& mess, string_view_type sv, size_type pos, size_type count)
57 {
58 if constexpr(sizeof(char) >= sizeof(Char))
59 {
60 mess.append(sv.substr(pos, count));
61 }
62 else
63 {
64 if(pos > sv.size()) throw std::out_of_range{"pos out of range"};
65 const auto end{count > sv.size() - pos ? sv.size() : pos + count};
66 std::transform(sv.begin() + pos, sv.begin() + end, std::back_inserter(mess), [](Char c) { return static_cast<char>(c); });
67 }

70 template<class Advisor>
71 static auto make_advisor(std::string_view info, string_view_type obtained, string_view_type prediction, size_type pos, const tutor<Advisor>& advisor)
72 {
73 constexpr size_type defaultOffset{30}, defaultCount{60}, npos{string_view_type::npos};
74 const auto sz{std::min(obtained.size(), prediction.size())};
75
76 auto newlineBackwards{ [pos](string_view_type sv){ return pos < sv.size() ? sv.rfind('\n', pos) : npos; } };
77
78 const auto loc{std::min(newlineBackwards(prediction), newlineBackwards(obtained))};
79
80 const size_type offset{loc < npos ? std::min(defaultOffset, pos - loc) : defaultOffset};
81
82 const auto lpos{pos < offset ? 0 :
83 pos < sz ? pos - offset : sz - std::min(sz, offset)};
84
85 struct message{ std::string mess; bool trunc{}; };
86
87 auto make{
88 [](string_view_type sv, size_type lpos) -> message {
89 std::string mess{lpos > 0 ? "..." : ""};
90
91 const bool newline{(lpos < sv.size()) && (sv[lpos] == '\n')};
92 if(newline) mess.append("\\n");
93
94 const auto rpos{sv.find('\n', lpos+1)};
95 const auto count{rpos == npos ? defaultCount : rpos - lpos};
96
97 if(newline && (count == 1))
98 {
99 mess.append("\\n");
100 }
101 else
102 {
103 if(newline) ++lpos;
104 appender(mess, sv, lpos, count);
105 }
106
107 const bool trunc{lpos + count < sv.size()};
108 if(trunc) mess.append("...");
109
110 return {mess, trunc};
111 }
112 };
113
114 const auto[obMess,obTrunc]{make(obtained, lpos)};
115 const auto[prMess,prTrunc]{make(prediction, lpos)};
116
117 const bool trunc{lpos > 0 || obTrunc || prTrunc};
118 const auto message{append_lines(info, trunc ? "Surrounding substring(s):" : "Full strings:",
119 prediction_message(obMess, prMess))};
120
121 if constexpr(std::invocable<tutor<Advisor>, Char, Char>)
122 {
123
124 return tutor{
125 [message, advisor] (Char a, Char b) {
126 auto m{message};
127 return append_advice(m, {advisor, a, b});
128 },
129 "\n"
130 };
131 }
132 else
133 {
134 return tutor{
135 [message](const auto&, const auto&) { return message; },
136 "\n"
137 };
138 }

142 template<test_mode Mode, class Advisor>
143 static void test(equality_check_t, test_logger<Mode>& logger, string_view_type obtained, string_view_type prediction, const tutor<Advisor>& advisor)
144 {
145 auto iters{std::mismatch(obtained.begin(), obtained.end(), prediction.begin(), prediction.end())};
146
147 if((iters.first != obtained.end()) && (iters.second != prediction.end()))
148 {
149 const auto dist{std::distance(obtained.begin(), iters.first)};
150 auto adv{make_advisor("", obtained, prediction, dist, advisor)};
151
152 const auto numLines{std::count(prediction.begin(), iters.second, '\n')};
153
154 const auto mess{
155 [dist,numLines]() {
156 std::string m{"First difference detected "};
157 numLines > 0 ? m.append("on line ").append(std::to_string(numLines+1))
158 : m.append("at character ").append(std::to_string(dist));
159
160 return m.append(":");
161 }()
162 };
163
164 check(equality, mess, logger, *(iters.first), *(iters.second), adv);
165 }
166 else if((iters.first != obtained.end()) || (iters.second != prediction.end()))
167 {
168 auto checker{
169 [&logger, obtained, prediction, &advisor](auto begin, auto iter, std::string_view state, std::string_view adjective){
170 const auto dist{std::distance(begin, iter)};
171 const auto info{std::string{"First "}.append(state).append(" character: ").append(display_character(*iter))};
172 auto adv{make_advisor(info, obtained, prediction, dist, advisor)};
173
174 const auto mess{append_lines("Lengths differ", std::string{"Obtained string is too "}.append(adjective))};
175
176 check(equality, mess, logger, obtained.size(), prediction.size(), adv);
177 }
178 };
179
180 if(iters.second != prediction.end())
181 {
182 checker(prediction.begin(), iters.second, "missing", "short");
183 }
184 else if(iters.first != obtained.end())
185 {
186 checker(obtained.begin(), iters.first, "excess", "long");
187 }
188 }

374 template<test_mode Mode, class Customization, invocable_r<bool, std::filesystem::path, std::filesystem::path> FinalTokenComparison>
375 static void check_path(test_logger<Mode>& logger, const Customization& custom, const std::filesystem::path& path, const std::filesystem::path& prediction, FinalTokenComparison compare)
376 {
377 namespace fs = std::filesystem;
378
379 const auto pathType{fs::status(path).type()};
380 const auto predictionType{fs::status(prediction).type()};
381
382 if(check(equality, path_check_preamble("Path type", path, prediction), logger, pathType, predictionType))
383 {
384 if(!path.empty())
385 {
386 const auto pathFinalToken{*(--path.end())};
387 const auto predictionFinalToken{*(--prediction.end())};
388 if(compare(pathFinalToken, predictionFinalToken))
389 {
390 switch(pathType)
391 {
392 case fs::file_type::regular:
393 check_file(logger, custom, path, prediction);
394 break;
395 case fs::file_type::directory:
396 check_directory(logger, custom, path, prediction, compare);
397 break;
398 default:
399 throw std::logic_error{std::string{"Detailed equivalance check for paths of type '"}
400 .append(serializer<fs::file_type>::make(pathType)).append("' not currently implemented")};
401 }
402 }
403 }
404 }

407 template<test_mode Mode, class Customization, invocable_r<bool, std::filesystem::path, std::filesystem::path> FinalTokenComparison>
408 static void check_directory(test_logger<Mode>& logger, const Customization& custom, const std::filesystem::path& dir, const std::filesystem::path& prediction, FinalTokenComparison compare)
409 {
410 namespace fs = std::filesystem;
411
412 auto generator{
413 [](const fs::path& dir) {
414 std::vector<fs::path> paths{};
415 for(const auto& p : fs::directory_iterator(dir))
416 {
417 if(std::find(excluded_files.begin(), excluded_files.end(), p.path().filename()) == excluded_files.end()
418 && std::find(excluded_extensions.begin(), excluded_extensions.end(), p.path().extension()) == excluded_extensions.end())
419 {
420 paths.push_back(p);
421 }
422 }
423
424 std::sort(paths.begin(), paths.end());
425
426 return paths;
427 }
428 };
429
430 const std::vector<fs::path> paths{generator(dir)}, predictedPaths{generator(prediction)};
431
432 check(equality, std::string{"Number of directory entries for "}.append(dir.generic_string()),
433 logger,
434 paths.size(),
435 predictedPaths.size());
436
437 const auto iters{std::mismatch(paths.begin(), paths.end(), predictedPaths.begin(), predictedPaths.end(),
438 [&dir,&prediction](const fs::path& lhs, const fs::path& rhs) {
439 return fs::relative(lhs, dir) == fs::relative(rhs, prediction);
440 })};
441 if((iters.first != paths.end()) && (iters.second != predictedPaths.end()))
442 {
443 check(equality, "First directory entry mismatch", logger, *iters.first, *iters.second);
444 }
445 else if(iters.first != paths.end())
446 {
447 check(equality, "First directory entry mismatch", logger, *iters.first, fs::path{});
448 }
449 else if(iters.second != predictedPaths.end())
450 {
451 check(equality, "First directory entry mismatch", logger, fs::path{}, *iters.second);
452 }
453 else
454 {
455 for(std::size_t i{}; i < paths.size(); ++i)
456 {
457 check_path(logger, custom, paths[i], predictedPaths[i], compare);
458 }
459 }
460 }
