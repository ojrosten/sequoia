48  template<class Advisor>
49  static auto make_advisor(std::string_view info, string_view_type obtained, string_view_type prediction, size_type pos, const tutor<Advisor>& advisor)
50  {
51  constexpr size_type defaultOffset{30}, defaultCount{60}, npos{string_view_type::npos};
52  const auto sz{std::min(obtained.size(), prediction.size())};
53 
54  auto newlineBackwards{ [pos](string_view_type sv){ return pos < sv.size() ? sv.rfind('\n', pos) : npos; } };
55 
56  const auto loc{std::min(newlineBackwards(prediction), newlineBackwards(obtained))};
57 
58  const size_type offset{loc < npos ? std::min(defaultOffset, pos - loc) : defaultOffset};
59 
60  const auto lpos{pos < offset ? 0 :
61  pos < sz ? pos - offset : sz - std::min(sz, offset)};
62 
63  struct message{ std::string mess; bool trunc{}; };
64 
65  auto make{
66  [lpos](string_view_type sv) -> message {
67  std::string mess{lpos > 0 ? "..." : ""};
68 
69  const auto pos{lpos < sv.size() && sv[lpos] == '\n' ? lpos + 1 : lpos};
70  if(pos == lpos + 1) mess.append("\\n");
71 
72  const auto rpos{sv.find('\n', pos+1)};
73  const auto count{rpos == npos ? defaultCount : rpos - pos};
74 
75  mess.append(sv.substr(pos, count));
76 
77  const bool trunc{lpos + count < sv.size()};
78  if(trunc) mess.append("...");
79 
80  return {mess, trunc};
81  }
82  };
83 
84  const auto[obMess,obTrunc]{make(obtained)};
85  const auto[prMess,prTrunc]{make(prediction)};
86 
87  const bool trunc{lpos > 0 || obTrunc || prTrunc};
88  const auto message{append_lines(info, trunc ? "Surrounding substring(s):" : "Full strings:",
89  prediction_message(obMess, prMess))};
90 
91  if constexpr(invocable<tutor<Advisor>, Char, Char>)
92  {
93 
94  return tutor{
95  [message, advisor] (Char a, Char b) {
96 
97  auto m{message};
98  return append_advice(m, {advisor, a, b});
99  },
100  "\n"
101  };
102  }
103  else
104  {
105  return tutor{
106  [message](Char, Char) { return message; },
107  "\n"
108  };
109  }
110  }

112  template<test_mode Mode, class Advisor>
113  static void check(test_logger<Mode>& logger, string_view_type obtained, string_view_type prediction, const tutor<Advisor>& advisor)
114  {
115  auto iters{std::mismatch(obtained.begin(), obtained.end(), prediction.begin(), prediction.end())};
116 
117  if((iters.first != obtained.end()) && (iters.second != prediction.end()))
118  {
119  const auto dist{std::distance(obtained.begin(), iters.first)};
120  auto adv{make_advisor("", obtained, prediction, dist, advisor)};
121 
122  const auto mess{
123  std::string{"First difference detected at character "}
124  .append(std::to_string(dist)).append(":")
125  };
126 
127  check_equality(mess, logger, *(iters.first), *(iters.second), adv);
128  }
129  else if((iters.first != obtained.end()) || (iters.second != prediction.end()))
130  {
131  auto check{
132  [&logger, obtained, prediction, &advisor](auto begin, auto iter, std::string_view state, std::string_view adjective){
133  const auto dist{std::distance(begin, iter)};
134  const auto info{std::string{"First "}.append(state).append(" character: ").append(display_character(*iter))};
135  auto adv{make_advisor(info, obtained, prediction, dist, advisor)};
136 
137  const auto mess{append_lines("Lengths differ", std::string{"Obtained string is too "}.append(adjective))};
138 
139  check_equality(mess, logger, obtained.size(), prediction.size(), adv);
140  }
141  };
142 
143  if(iters.second != prediction.end())
144  {
145  check(prediction.begin(), iters.second, "missing", "short");
146  }
147  else if(iters.first != obtained.end())
148  {
149  check(obtained.begin(), iters.first, "excess", "long");
150  }
151  }
152  }

312  static void check_directory(test_logger<Mode>& logger, const std::filesystem::path& dir, const std::filesystem::path& prediction)
313  {

315

319  for(const auto& p : fs::directory_iterator(dir))

323  }

328  }
329  };
