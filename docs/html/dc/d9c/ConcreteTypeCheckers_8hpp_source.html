53  template<class Advisor>
54  static auto make_advisor(std::string_view info, string_view_type obtained, string_view_type prediction, size_type pos, const tutor<Advisor>& advisor)
55  {
56  constexpr size_type defaultOffset{30}, defaultCount{60}, npos{string_view_type::npos};
57  const auto sz{std::min(obtained.size(), prediction.size())};
58 
59  auto newlineBackwards{ [pos](string_view_type sv){ return pos < sv.size() ? sv.rfind('\n', pos) : npos; } };
60 
61  const auto loc{std::min(newlineBackwards(prediction), newlineBackwards(obtained))};
62 
63  const size_type offset{loc < npos ? std::min(defaultOffset, pos - loc) : defaultOffset};
64 
65  const auto lpos{pos < offset ? 0 :
66  pos < sz ? pos - offset : sz - std::min(sz, offset)};
67 
68  struct message{ std::string mess; bool trunc{}; };
69 
70  auto make{
71  [lpos](string_view_type sv) -> message {
72  std::string mess{lpos > 0 ? "..." : ""};
73 
74  const auto pos{lpos < sv.size() && sv[lpos] == '\n' ? lpos + 1 : lpos};
75  if(pos == lpos + 1) mess.append("\\n");
76 
77  const auto rpos{sv.find('\n', pos+1)};
78  const auto count{rpos == npos ? defaultCount : rpos - pos};
79 
80  mess.append(sv.substr(pos, count));
81 
82  const bool trunc{lpos + count < sv.size()};
83  if(trunc) mess.append("...");
84 
85  return {mess, trunc};
86  }
87  };
88 
89  const auto[obMess,obTrunc]{make(obtained)};
90  const auto[prMess,prTrunc]{make(prediction)};
91 
92  const bool trunc{lpos > 0 || obTrunc || prTrunc};
93  const auto message{append_lines(info, trunc ? "Surrounding substring(s):" : "Full strings:",
94  prediction_message(obMess, prMess))};
95 
96  if constexpr(invocable<tutor<Advisor>, Char, Char>)
97  {
98 
99  return tutor{
100  [message, advisor] (Char a, Char b) {
101  auto m{message};
102  return append_advice(m, {advisor, a, b});
103  },
104  "\n"
105  };
106  }
107  else
108  {
109  return tutor{
110  [message](const auto&, const auto&) { return message; },
111  "\n"
112  };
113  }
114  }

116  template<test_mode Mode, class Advisor>
117  static void check(test_logger<Mode>& logger, string_view_type obtained, string_view_type prediction, const tutor<Advisor>& advisor)
118  {
119  auto iters{std::mismatch(obtained.begin(), obtained.end(), prediction.begin(), prediction.end())};
120 
121  if((iters.first != obtained.end()) && (iters.second != prediction.end()))
122  {
123  const auto dist{std::distance(obtained.begin(), iters.first)};
124  auto adv{make_advisor("", obtained, prediction, dist, advisor)};
125 
126  const auto mess{
127  std::string{"First difference detected at character "}
128  .append(std::to_string(dist)).append(":")
129  };
130 
131  check_equality(mess, logger, *(iters.first), *(iters.second), adv);
132  }
133  else if((iters.first != obtained.end()) || (iters.second != prediction.end()))
134  {
135  auto check{
136  [&logger, obtained, prediction, &advisor](auto begin, auto iter, std::string_view state, std::string_view adjective){
137  const auto dist{std::distance(begin, iter)};
138  const auto info{std::string{"First "}.append(state).append(" character: ").append(display_character(*iter))};
139  auto adv{make_advisor(info, obtained, prediction, dist, advisor)};
140 
141  const auto mess{append_lines("Lengths differ", std::string{"Obtained string is too "}.append(adjective))};
142 
143  check_equality(mess, logger, obtained.size(), prediction.size(), adv);
144  }
145  };
146 
147  if(iters.second != prediction.end())
148  {
149  check(prediction.begin(), iters.second, "missing", "short");
150  }
151  else if(iters.first != obtained.end())
152  {
153  check(obtained.begin(), iters.first, "excess", "long");
154  }
155  }
156  }
