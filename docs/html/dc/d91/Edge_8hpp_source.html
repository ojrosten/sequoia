50 class weighting

54
55 template<class... Args>
56 requires (sizeof...(Args) > 0)
57 constexpr void weight(Args&&... args)
58 {
59 WeightHandler::get(m_Weight) = weight_type{std::forward<Args>(args)...};
60 }
61
62 [[nodiscard]]
63 constexpr const weight_type& weight() const noexcept { return WeightHandler::get(m_Weight); }
64
65 template<std::invocable<weight_type&> Fn>

67 {

69 }

71 [[nodiscard]]
72 friend constexpr bool operator==(const weighting& lhs, const weighting& rhs) noexcept
