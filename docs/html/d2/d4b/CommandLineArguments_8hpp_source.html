141 template<std::invocable<std::string> Fn>
142 [[nodiscard]]
143 std::string parse_invoke_depth_first(int argc, char** argv, const options_forest options, Fn zerothArgProcessor)
144 {
145 auto[zerothArg, operations, help]{parse(argc, argv, options)};
146
147 if(help.empty())
148 {
149 zerothArgProcessor(zerothArg);
150
151 using namespace maths;
152 for(const auto& operTree : operations)
153 {
154 depth_first_search(operTree,
155 ignore_disconnected_t{},
156 [&operTree](const auto node) {
157 const operation& oper{operTree.cbegin_node_weights()[node]};
158 if(oper.early) oper.early(oper.arguments);
159 },
160 [&operTree](const auto node) {
161 const operation& oper{operTree.cbegin_node_weights()[node]};
162 if(oper.late) oper.late(oper.arguments);
163 });
164 }
165 }
166
167 return help;
168 }
