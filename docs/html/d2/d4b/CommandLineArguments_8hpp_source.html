107 template<std::invocable<std::string> Fn>
108 [[nodiscard]]
109 std::string parse_invoke_depth_first(int argc, char** argv, const options_forest options, Fn zerothArgProcessor)
110 {
111 auto[zerothArg, operations, help]{parse(argc, argv, options)};
112
113 if(help.empty())
114 {
115 zerothArgProcessor(zerothArg);
116
117 using namespace maths;
118 for(const auto& operTree : operations)
119 {
120 depth_first_search(operTree,
121 ignore_disconnected_t{},
122 [&operTree](const auto node) {
123 const operation& oper{operTree.cbegin_node_weights()[node]};
124 if(oper.early) oper.early(oper.arguments);
125 },
126 [&operTree](const auto node) {
127 const operation& oper{operTree.cbegin_node_weights()[node]};
128 if(oper.late) oper.late(oper.arguments);
129 });
130 }
131 }
132
133 return help;
134 }
