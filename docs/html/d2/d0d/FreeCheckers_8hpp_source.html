279 template<class CheckType, test_mode Mode, class T, class S, class... U>
280 requires implements_general_equivalence_check<Mode, CheckType, T, S, U...>
281 bool general_equivalence_check(CheckType flavour, std::string description, test_logger<Mode>& logger, const T& obtained, const S& s, const U&... u)
282 {
283 using processor = equivalent_type_processor<S, U...>;
284
285 const auto msg{
286 [flavour, &description, types{processor::info()}] () -> std::string&& {
287 return std::move(
288 append_lines(description,
289 "Comparison performed using:",
290 make_type_info<value_tester<T>>(),
291 std::format("Checking for {} with:", to_string(flavour)),
292 types).append("\n")
293 );
294 }
295 };
296
297 sentinel<Mode> sentry{logger, msg()};
298
299 if constexpr(tester_for<CheckType, Mode, T, T, S, U...>)
300 {
301 value_tester<T>::test(flavour, logger, obtained, s, u...);
302 }
303 else if constexpr(processor::ends_with_tutor)
304 {
305 using delegator = equivalence_checker_delegator<T>;
306
307 if constexpr(impl::invocable_without_last_arg<delegator, CheckType, test_logger<Mode>&, T, S, U...>)
308 {
309 invoke_with_specified_args(delegator{}, std::make_index_sequence<sizeof...(U) + 3>(), flavour, logger, obtained, s, u...);
310 }
311 else
312 {
313 static_assert(dependent_false<value_tester<T>>::value, "Should never be triggered; indicates mismatch between requirement and logic");
314 }
315 }
316 else if constexpr(tester_for<CheckType, Mode, T, T, S, U..., tutor<null_advisor>>)
317 {
318 value_tester<T>::test(flavour, logger, obtained, s, u..., tutor<null_advisor>{});
319 }
320 else
321 {
322 static_assert(dependent_false<value_tester<T>>::value, "Should never be triggered; indicates mismatch between requirement and logic");
323 }
324
325 return !sentry.failure_detected();
326 }

329 requires (std::invocable<Compare, T, T> && (!IsFinalMessage || reportable<T>) && !std::is_array_v<T>)
330 void binary_comparison(final_message_constant<IsFinalMessage>, sentinel<Mode>& sentry, Compare compare, const T& obtained, const T& prediction, tutor<Advisor> advisor)

339 }

379 {
380 [[nodiscard]]
381 std::string operator()(std::string mess) const
382 {
383 return mess;
384 }
385 };

387 template
388 <
389 class E,
390 test_mode Mode,
391 class Fn,
392 invocable_r<std::string, std::string> Postprocessor=default_exception_message_postprocessor
393 >
394 bool check_exception_thrown(std::string description, test_logger<Mode>& logger, Fn&& function, Postprocessor postprocessor={})
395 {
396 auto message{
397 [&description]() -> std::string&& {
398 return std::move(append_lines(description, "Expected Exception Type:", make_type_info<E>()));
399 }()
400 };
401
402 sentinel<Mode> sentry{logger, message};
403 sentry.log_check();
404 try
405 {
406 function();
407 sentry.log_failure("No exception thrown");
408 return false;
409 }
410 catch(const E& e)
411 {
412 sentry.log_caught_exception_message(postprocessor(exception_message_extractor<E>::get(e)));
413 return true;
414 }
415 catch(const std::exception& e)
416 {
417 std::string msg{append_lines("Unexpected exception thrown (caught by std::exception&):", "\"").append(e.what()).append("\"\n")};
418
419 sentry.log_failure(msg);
420 return false;
421 }
422 catch(...)
423 {
424 sentry.log_failure("Unknown exception thrown\n");
425 return false;
426 }
427 }

435 template
436 <
437 class CheckType,
438 test_mode Mode,
439 std::input_or_output_iterator Iter,
440 std::sentinel_for<Iter> Sentinel,
441 std::input_or_output_iterator PredictionIter,
442 std::sentinel_for<PredictionIter> PredictionSentinel,
443 class Advisor = null_advisor
444 >
445 bool check(CheckType flavour,
446 std::string description,
447 test_logger<Mode>& logger,
448 Iter first,
449 Sentinel last,
450 PredictionIter predictionFirst,
451 PredictionSentinel predictionLast,
452 tutor<Advisor> advisor = {})
453 {
454 auto info{
455 [&description]() -> std::string&& {
456 return description.empty() || description.back() == '\n' ? std::move(description) : std::move(description.append("\n"));
457 }
458 };
459
460 sentinel<Mode> sentry{logger, info()};
461 bool equal{true};
462
463 const auto actualSize{fixed_width_unsigned_cast(std::ranges::distance(first, last))};
464 const auto predictedSize{fixed_width_unsigned_cast(std::ranges::distance(predictionFirst, predictionLast))};
465 if(check(equality, "Container size wrong", logger, actualSize, predictedSize))
466 {
467 auto predictionIter{predictionFirst};
468 auto iter{first};
469 for(; predictionIter != predictionLast; std::ranges::advance(predictionIter, 1), std::ranges::advance(iter, 1))
470 {
471 const auto dist{std::ranges::distance(predictionFirst, predictionIter)};
472 auto mess{std::format("Element {} of range incorrect", dist)};
473 if(!check(flavour, std::move(mess), logger, *iter, *predictionIter, advisor)) equal = false;
474 }
475 }
476 else
477 {
478 equal = false;
479 }
480
481 return equal;
482 }

489 template<class Compare, test_mode Mode, class T, class Advisor=null_advisor>
490 requires maybe_comparison_type<Compare, T>
491 bool check(Compare compare,
492 std::string description,
493 test_logger<Mode>& logger,
494 const T& obtained,
495 const T& prediction,
496 tutor<Advisor> advisor={})
497 {
498 sentinel<Mode> sentry{logger, add_type_info<T>(std::move(description))};
499
500 if constexpr(std::invocable<Compare, T, T>)
501 {
502 using finality = final_message_constant<!faithful_range<T>>;
503 binary_comparison(finality{}, sentry, std::move(compare), obtained, prediction, advisor);
504 }
505 else
506 {
507 check(std::move(compare), "", logger, obtained.begin(), obtained.end(), prediction.begin(), prediction.end(), advisor);
508 }
509
510 return !sentry.failure_detected();
511 }
