278  requires implements_general_equivalence_check<Mode, CheckType, T, S, U...>
279  bool general_equivalence_check(CheckType flavour, std::string_view description, test_logger<Mode>& logger, const T& obtained, const S& s, const U&... u)
280  {
281  using processor = equivalent_type_processor<S, U...>;
282 
283  const auto msg{
284  [flavour, description, types{processor::info()}] (){
285  return append_lines(description,
286  "Comparison performed using:",
287  make_type_info<value_tester<T>>(),
288  std::string{"Checking for "}.append(to_string(flavour)).append(" with:"),
289  types)
290  .append("\n");
291  }()
292  };
293 
294  sentinel<Mode> sentry{logger, msg};
295 
296  if constexpr(tester_for<CheckType, Mode, T, T, S, U...>)
297  {
298  value_tester<T>::test(flavour, logger, obtained, s, u...);
299  }
300  else if constexpr(processor::ends_with_tutor)
301  {
302  using delegator = equivalence_checker_delegator<T>;
303 
304  if constexpr(impl::invocable_without_last_arg<delegator, CheckType, test_logger<Mode>&, T, S, U...>)
305  {
306  invoke_with_specified_args(delegator{}, std::make_index_sequence<sizeof...(U) + 3>(), flavour, logger, obtained, s, u...);
307  }
308  else
309  {
310  static_assert(dependent_false<value_tester<T>>::value, "Should never be triggered; indicates mismatch between requirement and logic");
311  }
312  }
313  else if constexpr(tester_for<CheckType, Mode, T, T, S, U..., tutor<null_advisor>>)
314  {
315  value_tester<T>::test(flavour, logger, obtained, s, u..., tutor<null_advisor>{});
316  }
317  else
318  {
319  static_assert(dependent_false<value_tester<T>>::value, "Should never be triggered; indicates mismatch between requirement and logic");
320  }
321 
322  return !sentry.failure_detected();
323  }

442  bool check(CheckType flavour,

444  test_logger<Mode>& logger,
445  Iter first,
446  Sentinel last,
447  PredictionIter predictionFirst,
448  PredictionSentinel predictionLast,
449  tutor<Advisor> advisor = {})
450  {
451  auto info{

454  ? std::string{description} : std::string{description}.append("\n");

458  sentinel<Mode> sentry{logger, info()};
459  bool equal{true};
460 
461  using std::distance;
462  using std::advance;

464  const auto actualSize{fixed_width_unsigned_cast(distance(first, last))};
465  const auto predictedSize{fixed_width_unsigned_cast(distance(predictionFirst, predictionLast))};
466  if(check(equality, "Container size wrong", logger, actualSize, predictedSize))

469  auto iter{first};
470  for(; predictionIter != predictionLast; advance(predictionIter, 1), advance(iter, 1))

473  auto mess{("Element " + std::to_string(dist)).append(" of range incorrect")};

475  }

479  equal = false;
480  }
481
