219  template<class EquivChecker, test_mode Mode, class Customization, class T, class S, class... U>
220  requires implements_general_equivalence_check<EquivChecker, Mode, Customization, T, S, U...>
221  bool general_equivalence_check(std::string_view description, test_logger<Mode>& logger, [[maybe_unused]] const value_based_customization<Customization>& customization, const T& value, const S& s, const U&... u)
222  {
223  using processor = equivalent_type_processor<S, U...>;
224 
225  const auto msg{
226  [description, types{processor::info()}] (){
227  return append_lines(description, "Comparison performed using:", make_type_info<EquivChecker>(), "With equivalent types:", types)
228  .append("\n");
229  }()
230  };
231 
232  sentinel<Mode> sentry{logger, msg};
233 
234  if constexpr(checker_for<EquivChecker, Mode, Customization, T, S, U...>)
235  {
236  EquivChecker::check(logger, customization.customizer, value, s, u...);
237  }
238  else if constexpr(checker_for<EquivChecker, Mode, T, S, U...>)
239  {
240  EquivChecker::check(logger, value, s, u...);
241  }
242  else if constexpr(processor::ends_with_tutor)
243  {
244  using delegator = equivalence_checker_delegator<EquivChecker>;
245 
246  if constexpr(impl::invocable_without_last_arg<delegator, test_logger<Mode>&, Customization, T, S, U...>)
247  {
248  invoke_with_specified_args(delegator{}, std::make_index_sequence<sizeof...(U) + 3>(), logger, customization.customizer, value, s, u...);
249  }
250  else if constexpr(impl::invocable_without_last_arg<delegator, test_logger<Mode>&, T, S, U...>)
251  {
252  invoke_with_specified_args(delegator{}, std::make_index_sequence<sizeof...(U) + 2>(), logger, value, s, u...);
253  }
254  else
255  {
256  static_assert(dependent_false<EquivChecker>::value, "Should never be triggered; indicates mismatch between requirement and logic");
257  }
258  }
259  else
260  {
261  if constexpr(checker_for<EquivChecker, Mode, Customization, T, S, U..., tutor<null_advisor>>)
262  {
263  EquivChecker::check(logger, customization.customizer, value, s, u..., tutor<null_advisor>{});
264  }
265  else if constexpr(checker_for<EquivChecker, Mode, T, S, U..., tutor<null_advisor>>)
266  {
267  EquivChecker::check(logger, value, s, u..., tutor<null_advisor>{});
268  }
269  else
270  {
271  static_assert(dependent_false<EquivChecker>::value, "Should never be triggered; indicates mismatch between requirement and logic");
272  }
273  }
274 
275  return !sentry.failure_detected();
276  }

416  <
417  test_mode Mode,
418  class ElementDispatchDiscriminator,
419  class Customization,
420  std::input_or_output_iterator Iter,
421  std::sentinel_for<Iter> Sentinel,
422  std::input_or_output_iterator PredictionIter,
423  std::sentinel_for<PredictionIter> PredictionSentinel,
424  class Advisor=null_advisor
425  >
426  bool check_range(std::string_view description,
427  test_logger<Mode>& logger,
428  ElementDispatchDiscriminator discriminator,
429  const value_based_customization<Customization>& customization,
430  Iter first,
431  Sentinel last,
432  PredictionIter predictionFirst,
433  PredictionSentinel predictionLast,
434  tutor<Advisor> advisor={})
435  {
436  auto info{
437  [description](){
438  return description.empty() || description.back() == '\n'
439  ? std::string{description} : std::string{description}.append("\n");
440  }
441  };
442 
443  sentinel<Mode> sentry{logger, info()};
444  bool equal{true};
445 
446  using std::distance;
447  using std::advance;
448 
449  const auto actualSize{fixed_width_unsigned_cast(distance(first, last))};
450  const auto predictedSize{fixed_width_unsigned_cast(distance(predictionFirst, predictionLast))};
451  if(check_equality("Container size wrong", logger, actualSize, predictedSize))
452  {
453  auto predictionIter{predictionFirst};
454  auto iter{first};
455  for(; predictionIter != predictionLast; advance(predictionIter, 1), advance(iter, 1))
456  {
457  const auto dist{distance(predictionFirst, predictionIter)};
458  auto mess{("Element " + std::to_string(dist)).append(" of range incorrect")};
459  if(!dispatch_check(std::move(mess), logger, discriminator, customization, *iter, *predictionIter, advisor)) equal = false;
460  }
461  }
462  else
463  {
464  equal = false;
465  }
466 
467  return equal;
468  }
