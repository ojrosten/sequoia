126  bool general_equivalence_check(std::string_view description, test_logger<Mode>& logger, const T& value, const S& s, const U&... u)
127  {
128  constexpr equivalent_type_processor<S, U...> processor{};
129 
130  const auto msg{
131  [description, types{processor.info()}](){
132  return append_lines(description, "Comparison performed using:", make_type_info<EquivChecker>(), "With equivalent types:", types)
133  .append("\n");
134  }()
135  };
136 
137  sentinel<Mode> sentry{logger, msg};
138 
139  if constexpr(checker_for<EquivChecker, Mode, T, S, U...>)
140  {
141  EquivChecker::check(logger, value, s, u...);
142  }
143  else if constexpr(processor.ends_with_tutor)
144  {
145  auto fn{
146  [&logger,&value](auto&&... predictions){
147  EquivChecker::check(logger, value, std::forward<decltype(predictions)>(predictions)...);
148  }
149  };
150 
151  invoke_with_specified_args(fn, std::make_index_sequence<sizeof...(u)>(), s, u...);
152  }
153  else

155  EquivChecker::check(logger, value, s, u..., tutor<null_advisor>{});
156  }
157 
158  return !sentry.failure_detected();

186  template<test_mode Mode, class T, class Advisor=null_advisor>
187  bool dispatch_check(std::string_view description, test_logger<Mode>& logger, equality_tag, const T& obtained, const T& prediction, [[maybe_unused]] tutor<Advisor> advisor=tutor<Advisor>{})
188  {
189  constexpr bool delegate{has_detailed_equality_checker_v<T> || range<T>};
190 
191  static_assert(delegate || equality_comparable<T>,
192  "Provide either a specialization of detailed_equality_checker or"
193  "ensure operator== and != exists, together with a specialization of serializer");
194 
195  sentinel<Mode> sentry{logger, add_type_info<T>(description)};
196 
197  if constexpr(equality_comparable<T>)
198  {
199  sentry.log_check();
200  if(!(prediction == obtained))
201  {
202  auto message{failure_message(obtained, prediction)};
203 
204  append_advice(message, {advisor, obtained, prediction});
205 
206  sentry.log_failure(std::move(message));
207  }
208  }
209 
210  if constexpr(delegate)
211  {
212  if constexpr(has_detailed_equality_checker_v<T>)
213  {
214  if constexpr(checker_for<detailed_equality_checker<T>, Mode, T, T, tutor<Advisor>>)
215  {
216  detailed_equality_checker<T>::check(logger, obtained, prediction, advisor);
217  }
218  else
219  {
220  detailed_equality_checker<T>::check(logger, obtained, prediction);
221  }
222  }
223  else if constexpr(range<T>)
224  {
225  check_range("", logger, std::begin(obtained), std::end(obtained), std::begin(prediction), std::end(prediction), advisor);
226  }
227  else
228  {
229  static_assert(dependent_false<T>::value, "Unable to check detailed equality for Type");
230  }
231  }
232 
233  return !sentry.failure_detected();
234  }

287  bool check_range(std::string_view description, test_logger<Mode>& logger, ElementDispatchDiscriminator discriminator, Iter first, Iter last, PredictionIter predictionFirst, PredictionIter predictionLast, tutor<Advisor> advisor=tutor<Advisor>{})
288  {

299  using std::distance;
300  using std::advance;
301

303  if(check_equality("Container size wrong", logger, distance(first, last), predictedSize))
304  {

307  for(; predictionIter != predictionLast; advance(predictionIter, 1), advance(iter, 1))
