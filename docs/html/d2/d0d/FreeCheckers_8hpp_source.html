203  template<test_mode Mode, class T, class Advisor=null_advisor>
204  bool dispatch_check(std::string_view description, test_logger<Mode>& logger, equality_tag, const T& obtained, const T& prediction, [[maybe_unused]] tutor<Advisor> advisor=tutor<Advisor>{})
205  {
206  constexpr bool delegate{has_detailed_equality_checker_v<T> || range<T>};
207 
208  static_assert(delegate || equality_comparable<T>,
209  "Provide either a specialization of detailed_equality_checker or"
210  "ensure operator== and != exists, together with a specialization of serializer");
211 
212  sentinel<Mode> sentry{logger, add_type_info<T>(description)};
213 
214  if constexpr(equality_comparable<T>)
215  {
216  sentry.log_check();
217  if(!(prediction == obtained))
218  {
219  auto message{failure_message(obtained, prediction)};
220 
221  append_advice(message, {advisor, obtained, prediction});
222 
223  sentry.log_failure(std::move(message));
224  }
225  }
226 
227  if constexpr(delegate)
228  {
229  if constexpr(has_detailed_equality_checker_v<T>)
230  {
231  if constexpr(checker_for<detailed_equality_checker<T>, Mode, T, T, tutor<Advisor>>)
232  {
233  detailed_equality_checker<T>::check(logger, obtained, prediction, advisor);
234  }
235  else
236  {
237  detailed_equality_checker<T>::check(logger, obtained, prediction);
238  }
239  }
240  else if constexpr(range<T>)
241  {
242  check_range("", logger, std::begin(obtained), std::end(obtained), std::begin(prediction), std::end(prediction), advisor);
243  }
244  else
245  {
246  static_assert(dependent_false<T>::value, "Unable to check detailed equality for Type");
247  }
248  }
249 
250  return !sentry.failure_detected();
251  }
