221  template<test_mode Mode, class T, class Advisor=null_advisor>
222  bool dispatch_check(std::string_view description, test_logger<Mode>& logger, equality_tag, const T& obtained, const T& prediction, [[maybe_unused]] tutor<Advisor> advisor={})
223  {
224  constexpr bool delegate{has_detailed_equality_checker_v<T> || range<T>};
225 
226  static_assert(delegate || equality_comparable<T>,
227  "Provide either a specialization of detailed_equality_checker or"
228  "ensure operator== and != exists, together with a specialization of serializer");
229 
230  sentinel<Mode> sentry{logger, add_type_info<T>(description)};
231 
232  if constexpr(equality_comparable<T>)
233  {
234  binary_comparison(sentry, std::equal_to<T>{}, obtained, prediction, advisor);
235  }
236 
237  if constexpr(delegate)
238  {
239  if constexpr(has_detailed_equality_checker_v<T>)
240  {
241  if constexpr(checker_for<detailed_equality_checker<T>, Mode, T, T, tutor<Advisor>>)
242  {
243  detailed_equality_checker<T>::check(logger, obtained, prediction, advisor);
244  }
245  else
246  {
247  detailed_equality_checker<T>::check(logger, obtained, prediction);
248  }
249  }
250  else if constexpr(range<T>)
251  {
252  check_range("", logger, std::begin(obtained), std::end(obtained), std::begin(prediction), std::end(prediction), advisor);
253  }
254  else
255  {
256  static_assert(dependent_false<T>::value, "Unable to check detailed equality for Type");
257  }
258  }
259 
260  return !sentry.failure_detected();
261  }
