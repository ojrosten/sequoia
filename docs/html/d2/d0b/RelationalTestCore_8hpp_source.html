61  template<test_mode Mode, class Compare, class T, class Advisor>
62  bool dispatch_check(std::string_view description, test_logger<Mode>& logger, relational_compare<Compare> c, const T& obtained, const T& prediction, tutor<Advisor> advisor)
63  {
64  sentinel<Mode> sentry{logger, add_type_info<T>(description)};
65 
66  if constexpr(invocable<Compare, T, T>)
67  {
68  sentry.log_check();
69  if(!c.compare(obtained, prediction))
70  {
71  std::string message{};
72  if constexpr(relational_reporter<Compare, T>)
73  {
74  message = c.compare.report(obtained, prediction);
75  }
76  else
77  {
78  message = append_lines("Relational comparison failed", prediction_message(to_string(obtained), to_string(prediction)));
79  }
80 
81  append_advice(message, {advisor, obtained, prediction});
82 
83  sentry.log_failure(message);
84  }
85 
86  return !sentry.failure_detected();
87  }
88  else if constexpr(range<T>)
89  {
90  return check_range("", logger, std::move(c), obtained.begin(), obtained.end(), prediction.begin(), prediction.end(), advisor);
91  }
92  else
93  {
94  static_assert(dependent_false<T>::value, "Compare cannot consume T directly nor interpret as a range");
95  }
96  }
