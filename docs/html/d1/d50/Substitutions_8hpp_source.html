110 template<invocable_r<bool, char> LeftPred, invocable_r<bool, char> RightPred>
111 std::string& replace_all(std::string& text, LeftPred lPred, std::string_view from, RightPred rPred, std::string_view to)
112 {
113 constexpr auto npos{std::string::npos};
114 std::string::size_type pos{};
115 while((pos = text.find(from, pos)) != npos)
116 {
117 if( (((pos > 0) && lPred(text[pos - 1])) || ((pos == 0) && lPred('\0')))
118 && ( ((pos + from.length() < text.length()) && rPred(text[pos + from.length()]))
119 || ((pos + from.length() == text.length()) && rPred('\0')))
120 )
121 {
122 text.replace(pos, from.length(), to);
123 pos += (to.length() + 1);
124 }
125 else
126 {
127 pos += (from.length() + 1) ;
128 }
129 }
130
131 return text;
132 }
