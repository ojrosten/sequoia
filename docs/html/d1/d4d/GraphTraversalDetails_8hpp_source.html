221 public:
222 template<std::forward_iterator Iter>
223 [[nodiscard]]
224 constexpr static bool loop_matched(Iter begin, Iter current)
225 {
226 for(auto i{begin}; i != current; ++i)
227 {
228 if(&(*i) == &(*current)) return true;
229 }
230
231 return false;
232 }

274 template
275 <
276 traversal_flavour F,
277 disconnected_discovery_mode FindDisconnected,
278 class NBEF,
279 class NAEF,
280 class EFTF,
281 class ESTF,
282 class TaskProcessingModel,
283 class... QArgs
284 >
285 requires (std::invocable<NBEF, edge_index_type> )
286 && (std::invocable<NAEF, edge_index_type> )
287 && (std::invocable<EFTF, const_edge_iterator>)
288 && (std::invocable<ESTF, const_edge_iterator>)
289 constexpr auto traverse(traversal_constant<F>,
290 const G& graph,
291 traversal_conditions<FindDisconnected> conditions,
292 NBEF&& nodeBeforeEdgesFn,
293 NAEF&& nodeAfterEdgesFn,
294 EFTF&& edgeFirstTraversalFn,
295 ESTF&& edgeSecondTraversalFn,
296 TaskProcessingModel&& taskProcessingModel,
297 QArgs&&... qargs)
298 {
299 // Note: do not forward any of the Fns as they could in principle end up repeatedly moved from.
300 // However, the Fns should not be captured by value as they may have mutable state with
301 // external visibility.
302
303 constexpr bool hasEdgeSecondFn{!std::same_as<std::remove_cvref_t<ESTF>, null_func_obj>};
304 static_assert(!directed(G::directedness) || !hasEdgeSecondFn,
305 "For a directed graph, edges are traversed only once: the edgeSecondTraversalFn is ignored and so should be the null_func_obj");
306
307 if(conditions.starting_index() < graph.order())
308 {
309 auto discovered{traversal_tracking_traits<G>::make_bitset(graph)};
310 auto processed{traversal_tracking_traits<G>::make_bitset(graph)};
311
312 auto nodeIndexQueue{traversal_traits<G, F, QArgs...>::make(std::forward<QArgs>(qargs)...)};
313
314 do
315 {
316 const auto restartNode{static_cast<edge_index_type>(conditions.compute_restart_index(discovered))};
317
318 nodeIndexQueue.push(restartNode);
319 conditions.register_discovered(discovered, restartNode);
320
321 while(!nodeIndexQueue.empty())
322 {
323 const auto nodeIndex{traversal_traits<G, F, QArgs...>::get_container_element(nodeIndexQueue)};
324 nodeIndexQueue.pop();
325
326 auto onDiscovery{[&nodeIndexQueue](const edge_index_type nextNode) { nodeIndexQueue.push(nextNode); }};
327
328 inner_loop(graph,
329 nodeIndex,
330 conditions,
331 traversal_traits<G, F, QArgs...>::begin(graph, nodeIndex),
332 traversal_traits<G, F, QArgs...>::end(graph, nodeIndex),
333 discovered,
334 processed,
335 onDiscovery,
336 nodeBeforeEdgesFn,
337 nodeAfterEdgesFn,
338 edgeFirstTraversalFn,
339 edgeSecondTraversalFn,
340 taskProcessingModel);
341 }
342 } while(!conditions.terminate(graph.order()));
343 }
344
345 return taskProcessingModel.get();
346 }

413 template
414 <
415 disconnected_discovery_mode FindDisconnected,
416 class Iter,
417 class Bitset,
418 class OnDiscovery,
419 class NBEF,
420 class NAEF,
421 class EFTF,
422 class ESTF,
423 class TaskProcessingModel
424 >
425 requires (std::invocable<NBEF, edge_index_type>)
426 && (std::invocable<NAEF, edge_index_type>)
427 && (std::invocable<EFTF, const_edge_iterator>)
428 && (std::invocable<ESTF, const_edge_iterator>)
429 && (std::invocable<OnDiscovery, edge_index_type> || std::same_as<OnDiscovery, recurse>)
430 constexpr void inner_loop([[maybe_unused]] const G& graph,
431 const edge_index_type nodeIndex,
432 traversal_conditions<FindDisconnected>& conditions,
433 Iter begin,
434 Iter end,
435 Bitset& discovered,
436 Bitset& processed,
437 OnDiscovery onDiscovery,
438 NBEF&& nodeBeforeEdgesFn,
439 NAEF&& nodeAfterEdgesFn,
440 EFTF&& edgeFirstTraversalFn,
441 ESTF&& edgeSecondTraversalFn,
442 TaskProcessingModel&& taskProcessingModel)
443 {
444 constexpr bool hasNodeBeforeFn{!std::same_as<std::remove_cvref_t<NBEF>, null_func_obj>};
445 if constexpr(hasNodeBeforeFn)
446 {
447 taskProcessingModel.push(nodeBeforeEdgesFn, nodeIndex);
448 }
449
450 [[maybe_unused]] loop_processor<G> loops{};
451 for(auto iter{begin}; iter != end; ++iter)
452 {
453 const auto nextNode{iter->target_node()};
454 constexpr bool hasEdgeFirstFn{!std::same_as<std::remove_cvref_t<EFTF>, null_func_obj>};
455 constexpr bool isDFS{std::same_as<OnDiscovery, recurse>};
456
457 if constexpr(isDFS)
458 {
459 if constexpr(hasEdgeFirstFn)
460 {
461 if(!discovered[nextNode]) taskProcessingModel.push(edgeFirstTraversalFn, iter);
462 }
463 }
464 else if constexpr(G::flavour != graph_flavour::directed)
465 {
466 constexpr bool hasEdgeSecondFn{!std::same_as<std::remove_cvref_t<ESTF>, null_func_obj>};
467 if constexpr(G::flavour == graph_flavour::directed_embedded)
468 {
469 if(is_loop(iter, nodeIndex))
470 {
471 const bool loopMatched{loops.loop_matched(begin, iter)};
472 if((iter->inverted() && !loopMatched) || (!iter->inverted() && loopMatched)) continue;
473 }
474 else
475 {
476 if(iter->source_node() != nodeIndex) continue;
477 }
478
479 if constexpr(hasEdgeFirstFn)
480 {
481 taskProcessingModel.push(edgeFirstTraversalFn, iter);
482 }
483 }
484 else if constexpr(hasEdgeFirstFn || hasEdgeSecondFn)
485 {
486 const bool loopMatched{is_loop(iter, nodeIndex) && loops.loop_matched(begin, iter)};
487 const bool secondTraversal{processed[nextNode] || loopMatched};
488 if(secondTraversal)
489 {
490 if constexpr(hasEdgeSecondFn)
491 {
492 taskProcessingModel.push(edgeSecondTraversalFn, iter);
493 }
494 }
495 else
496 {
497 if constexpr(hasEdgeFirstFn)
498 {
499 taskProcessingModel.push(edgeFirstTraversalFn, iter);
500 }
501 }
502 }
503 }
504 else if constexpr(hasEdgeFirstFn)
505 {
506 taskProcessingModel.push(edgeFirstTraversalFn, iter);
507 }
508
509 if(!discovered[nextNode])
510 {
511 conditions.register_discovered(discovered, nextNode);
512 if constexpr(std::same_as<OnDiscovery, recurse>)
513 {
514 inner_loop(graph,
515 nextNode,
516 conditions,
517 graph.cbegin_edges(nextNode),
518 graph.cend_edges(nextNode),
519 discovered,
520 processed,
521 recurse{},
522 nodeBeforeEdgesFn,
523 nodeAfterEdgesFn,
524 edgeFirstTraversalFn,
525 edgeSecondTraversalFn,
526 taskProcessingModel);
527 }
528 else
529 {
530 onDiscovery(nextNode);
531 }
532 }
533 }
534
535 if constexpr(!std::same_as<std::remove_cvref_t<NAEF>, null_func_obj>)
536 {
537 taskProcessingModel.push(nodeAfterEdgesFn, nodeIndex);
538 }
539
540 processed[nodeIndex] = true;
541 }
